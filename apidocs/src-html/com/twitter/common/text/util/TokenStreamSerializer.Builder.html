017    package com.twitter.common.text.util;

019    import java.io.ByteArrayInputStream;
020    import java.io.ByteArrayOutputStream;
021    import java.io.DataInputStream;
022    import java.io.DataOutputStream;
023    import java.io.IOException;
024    import java.io.InputStream;
025    import java.io.OutputStream;
026    import java.util.List;
027    
028    import com.google.common.annotations.VisibleForTesting;
029    import com.google.common.base.Preconditions;
030    import com.google.common.collect.Lists;
031    
032    import org.apache.lucene.util.AttributeSource;
033    
034    import com.twitter.common.text.token.TokenStream;

036    /**
037     * Helper class to serialize a TokenStream into a byte array.
038     *
039     * A list of AttributeSerializers must be defined using the Builder, which serialize
040     * and deserialize individual attributes.
041     *
042     * The same TokenStreamSerializer should be used for serialization/de-serialization, as the order
043     * of the {@link AttributeSerializer}s must be consistent.
044     */
045    public class TokenStreamSerializer {
046      public static enum Version {
047        VERSION_1
048      }
049    
050      private static final Version CURRENT_VERSION = Version.VERSION_1;
051    
052      private final List<AttributeSerializer> attributeSerializers;
053    
054      private TokenStreamSerializer(List<AttributeSerializer> attributeSerializers) {
055        this.attributeSerializers = attributeSerializers;
056      }
057    
058      /**
059       * Serialize the given TokenStream into a byte array using the provided AttributeSerializer(s).
060       * Note that this method doesn't serialize the CharSequence of the TokenStream - the caller
061       * has to take care of serializing this if necessary.
062       */
063      public final byte[] serialize(final TokenStream tokenStream) throws IOException {
064        ByteArrayOutputStream baos = new ByteArrayOutputStream();
065        AttributeOutputStream output = new AttributeOutputStream(baos);
066    
067        for (AttributeSerializer serializer : attributeSerializers) {
068          serializer.initialize(tokenStream, CURRENT_VERSION);
069        }
070    
071        int numTokens = 0;
072    
073        while (tokenStream.incrementToken()) {
074          serializeAttributes(output);
075          numTokens++;
076        }
077    
078        output.flush();
079    
080        byte[] data = baos.toByteArray();
081        baos.close();
082        baos = new ByteArrayOutputStream(8 + data.length);
083        output = new AttributeOutputStream(baos);
084        output.writeVInt(CURRENT_VERSION.ordinal());
085        output.writeVInt(numTokens);
086        output.write(data);
087        output.flush();
088    
089        return baos.toByteArray();
090      };
091    
092    
093      /**
094       * Deserializes the previously serialized TokenStream using the provided AttributeSerializer(s).
095       *
096       * This method only deserializes all Attributes; the CharSequence instance containing the text
097       * must be provided separately.
098       */
099      public final TokenStream deserialize(final byte[] data,
100                                           final CharSequence charSequence) throws IOException {
101        Preconditions.checkNotNull(data);
102        Preconditions.checkState(data.length > 0);
103    
104        ByteArrayInputStream bais = new ByteArrayInputStream(data);
105        final AttributeInputStream input = new AttributeInputStream(bais);
106    
107        int ordinal = input.readVInt();
108        if (ordinal > CURRENT_VERSION.ordinal()) {
109          throw new IOException("Version of serialized data is newer than the version this serializer" +
110                                    "supports: " + ordinal + " > " + CURRENT_VERSION.ordinal());
111        }
112    
113        final Version version = Version.values()[ordinal];
114        final int numTokens = input.readVInt();
115    
116        TokenStream tokenStream = new TokenStream() {
117          CharSequence chars;
118          int token = 0;
119    
120          @Override public boolean incrementToken() {
121            if (token < numTokens) {
122              token++;
123              try {
124                deserializeAttributes(input, chars);
125              } catch (IOException e) {
126                throw new RuntimeException(e);
127              }
128              return true;
129            }
130    
131            return false;
132          }
133    
134          @Override
135          public void reset(CharSequence input) {
136            chars = input;
137            token = 0;
138          }
139        };
140    
141        for (AttributeSerializer deserializer : attributeSerializers) {
142          deserializer.initialize(tokenStream, version);
143        }
144    
145        input.close();
146    
147        tokenStream.reset(charSequence);
148        return tokenStream;
149      };
150    
151      private void deserializeAttributes(AttributeInputStream input,
152                                         CharSequence charSequence) throws IOException {
153        for (AttributeSerializer serializer : attributeSerializers) {
154          serializer.deserialize(input, charSequence);
155        }
156      }
157    
158    
159      private void serializeAttributes(AttributeOutputStream output) throws IOException {
160        for (AttributeSerializer serializer : attributeSerializers) {
161          serializer.serialize(output);
162        }
163      }
164    
165      /**
166       * Returns a new Builder to build a TokenStreamSerializer.
167       */
168      public static Builder builder() {
169        return new Builder();
170      }
171    
172      /**
173       * Defines how individual attributes a (de)serialized.
174       */
175      public interface AttributeSerializer {
176        /**
177         * Initialises this AttributeSerializer. This method should be used to get the attribute
178         * instance from the TokenStream that this serializer handles. E.g.:
179         *
180         * CharSequenceTermAttribute termAtt =
181         *                    attributeSource.addAttribute(CharSequenceTermAttribute.class);
182         *
183         */
184        public abstract void initialize(AttributeSource attributeSource, Version version)
185            throws IOException;
186    
187        /**
188         * Serializes a single attribute.
189         */
190        public abstract void serialize(AttributeOutputStream output) throws IOException ;
191    
192        /**
193         * Deserializes a single attribute.
194         */
195        public abstract void deserialize(AttributeInputStream input, CharSequence charSequence)
196            throws IOException;
197      }
198    
199      /**
200       * Builds an TokenStreamSerializer.
201       */
202      public final static class Builder {
203        private final List<AttributeSerializer> attributeSerializers = Lists.newLinkedList();
204    
205        /**
206         * Adds an AttributeSerializer. The order in which the AttributeSerializers are added here
207         * is the same order in which they will be called for serializing a Token.
208         */
209        public Builder add(AttributeSerializer serializer) {
210          attributeSerializers.add(serializer);
211          return this;
212        }
213    
214        /**
215         * Builds the TokenStreamSerializer.
216         */
217        public TokenStreamSerializer build() {
218          return new TokenStreamSerializer(attributeSerializers);
219        }
220      }
221    
222      /**
223       * A DataOutputStream that supports VInt-encoding.
224       */
225      public static class AttributeOutputStream extends DataOutputStream {
226        @VisibleForTesting
227        AttributeOutputStream(OutputStream output) {
228          super(output);
229        }
230    
231        /**
232         * Writes a value using VInt encoding.
233         */
234        public final void writeVInt(int value) throws IOException {
235          while ((value & ~0x7F) != 0) {
236            writeByte((byte)((value & 0x7f) | 0x80));
237            value >>>= 7;
238          }
239          writeByte((byte)value);
240        }
241      }
242    
243      /**
244       * A DataInputStream that supports VInt-encoding.
245       */
246      public static class AttributeInputStream extends DataInputStream {
247        @VisibleForTesting
248        AttributeInputStream(InputStream input) {
249          super(input);
250        }
251    
252        /**
253         * Reads a value using VInt encoding.
254         */
255        public final int readVInt() throws IOException {
256          byte b = readByte();
257          int value = b & 0x7F;
258          for (int shift = 7; (b & 0x80) != 0; shift += 7) {
259            b = readByte();
260            value |= (b & 0x7F) << shift;
261          }
262          return value;
263        }
264      }
