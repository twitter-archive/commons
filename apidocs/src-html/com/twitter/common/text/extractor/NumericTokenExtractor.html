001    package com.twitter.common.text.extractor;
002    
003    import java.util.regex.Pattern;
004    
005    
006    public class NumericTokenExtractor extends RegexExtractor {
007      private static final String NUMERIC_CHARS = "[0-9０−９]+";
008      private static final String NUMERIC_CONNECTOR_CHARS = "[.．,，/]";
009      public static final Pattern NUMERIC_TOKEN_PATTERN =
010          Pattern.compile(NUMERIC_CHARS + "(?:(" + NUMERIC_CONNECTOR_CHARS + ")" + NUMERIC_CHARS + ")"
011              + "(?:\\1" + NUMERIC_CHARS + ")*");
012    
013      public NumericTokenExtractor() {
014        setRegexPattern(NUMERIC_TOKEN_PATTERN);
015      }
016    }
