001    package com.twitter.common.text;
002    
003    import com.google.common.base.Preconditions;
004    import com.twitter.common.text.combiner.PunctuationExceptionCombiner;
005    import com.twitter.common.text.token.TokenStream;
006    import com.twitter.common.text.token.TokenizedCharSequence;
007    import com.twitter.common.text.token.TokenizedCharSequenceStream;
008    import com.twitter.common.text.tokenizer.LatinTokenizer;
009    
010    import java.util.List;
011    
012    public abstract class TextTokenizer {
013      protected TokenStream tokenizationStream =
014          new TokenizedCharSequenceStream(applyDefaultChain(
015            new PunctuationExceptionCombiner.Builder(
016              new LatinTokenizer.Builder().build()).build()));
017    
018      public abstract TokenStream applyDefaultChain(TokenStream tokenizer);
019    
020      /**
021       * Returns {@code TokenStream} to tokenize a text.
022       *
023       * @return {@code TokenStream} to tokenize the text
024       */
025      public TokenStream getDefaultTokenStream() {
026        return tokenizationStream;
027      }
028    
029      /**
030       * Tokenizes a {@code CharSequence}, and returns a {@code TokenizedCharSequence} as a result.
031       *
032       * @param input text to be tokenized
033       * @return {@code TokenizedCharSequence} instance
034       */
035      public TokenizedCharSequence tokenize(CharSequence input) {
036        Preconditions.checkNotNull(input);
037        return TokenizedCharSequence.createFrom(input, getDefaultTokenStream());
038      }
039    
040      /**
041       * Tokenizes a {@code CharSequence} into a list of Strings.
042       *
043       * @param input text to be tokenized
044       * @return a list of tokens as String objects
045       */
046      public List<String> tokenizeToStrings(CharSequence input) {
047        Preconditions.checkNotNull(input);
048        TokenStream tokenizer = getDefaultTokenStream();
049        tokenizer.reset(input);
050        return tokenizer.toStringList();
051      }
052    }
