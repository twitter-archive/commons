050    /**
051     * Parsers for different supported argument types.
052     *
053     * TODO(William Farner): Add a mechanism for global registration of custom parsers.
054     *
055     * @author William Farner
056     */
057    public final class Parsers {
058    
059      private Parsers() {
060        // Utility
061      }
062    
063      /**
064       * Gets the parser associated with a class.
065       *
066       * @param cls Class to get the parser for.
067       * @return The parser for {@code cls} or {@code null} if no parser was found for that type.
068       */
069      public static Parser get(Class<?> cls) {
070        return REGISTRY.get(cls);
071      }
072    
073      /**
074       * Gets the parser associatd with a class.
075       *
076       * @param cls Class to get the parser for.
077       * @return The parser for {@code cls} or {@code null} if no parser was found for that type.
078       * @throws IllegalArgumentException If no parser was found for {@code cls}.
079       */
080      public static Parser checkedGet(Class<?> cls) throws IllegalArgumentException {
081        Parser parser = get(cls);
082        checkArgument(parser != null, "No parser found for class " + cls);
083        return parser;
084      }
085    
086      public static interface Parser<T> {
087        T parse(Type type, String raw) throws IllegalArgumentException;
088    
089        Class<T> getParsedClass();
090      }
091    
092      private static final Set<Class<? extends Parser<?>>> PARSER_CLASSES =
093          ImmutableSet.<Class<? extends Parser<?>>>builder()
094          .add(AmountParser.class)
095          .add(BooleanParser.class)
096          .add(ByteParser.class)
097          .add(CharacterParser.class)
098          .add(ClassParser.class)
099          .add(DateParser.class)
100          .add(DoubleParser.class)
101          .add(FileParser.class)
102          .add(FloatParser.class)
103          .add(InetSocketAddressParser.class)
104          .add(IntegerParser.class)
105          .add(ListParser.class)
106          .add(LongParser.class)
107          .add(MapParser.class)
108          .add(PairParser.class)
109          .add(SetParser.class)
110          .add(ShortParser.class)
111          .add(StringParser.class)
112          .add(UnitParser.class)
113          .add(URIParser.class)
114          .add(URLParser.class)
115          .build();
116      private static final ImmutableMap<Class<?>, Parser> REGISTRY;
117      static {
118        ImmutableMap.Builder<Class<?>, Parser> registryBuilder =
119          ImmutableMap.builder();
120        for (Class<? extends Parser<?>> parserClass : PARSER_CLASSES) {
121          try {
122            Parser parser = parserClass.newInstance();
123            registryBuilder.put(parser.getParsedClass(), parser);
124          } catch (InstantiationException e) {
125            throw new RuntimeException("Failed to instantiate parser class " + parserClass, e);
126          } catch (IllegalAccessException e) {
127            throw new RuntimeException("No access to constructor on parser " + parserClass, e);
128          }
129        }
130        REGISTRY = registryBuilder.build();
131      }
132    
133      public static final Splitter MULTI_VALUE_SPLITTER = Splitter.on(",")
134          .trimResults().omitEmptyStrings();
135    }
