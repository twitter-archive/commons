018    package com.twitter.common.args.parsers;
019    
020    /**
021     * Parser that handles common functionality for parsing numbers.
022     *
023     * @author William Farner
024     */
025    public abstract class NumberParser<T extends Number> extends NonParameterizedTypeParser<T> {
026    
027      /**
028       * Creates a new number parser.
029       *
030       * @param cls Class that the parser parses into, used to show useful error messages.
031       */
032      NumberParser(Class<T> cls) {
033        super(cls);
034      }
035    
036      /**
037       * Performs the actual parsing of the value into the target type.
038       *
039       * @param raw Raw value to parse.
040       * @return The parsed value.
041       * @throws NumberFormatException If the raw value does not represent a valid number of the target
042       *    type.
043       */
044      abstract T parseNumber(String raw) throws NumberFormatException;
045    
046      @Override
047      public T doParse(String raw) throws IllegalArgumentException {
048        try {
049          return parseNumber(raw);
050        } catch (NumberFormatException e) {
051          throw new IllegalArgumentException(String.format("Invalid %s value: " + e.getMessage(),
052              parsedClass.getSimpleName()));
053        }
054      }
055    }
