017    package com.twitter.common.args.parsers;
018    
019    import java.lang.reflect.Type;
020    import java.util.List;
021    
022    import com.google.common.base.Preconditions;
023    
024    import com.twitter.common.args.ArgParser;
025    import com.twitter.common.args.ParserOracle;
026    import com.twitter.common.args.TypeUtil;
027    
028    /**
029     * Class parser.
030     *
031     * @author William Farner
032     */
033    @ArgParser
034    public class ClassParser extends TypeParameterizedParser<Class> {
035    
036      public ClassParser() {
037        super(1);
038      }
039    
040      @Override
041      public Class doParse(ParserOracle parserOracle, String raw, final List<Type> typeParams) {
042        Class<?> rawClassType = TypeUtil.getRawType(typeParams.get(0));
043        try {
044          Class<?> actualClass = Class.forName(raw);
045          Preconditions.checkArgument(rawClassType.isAssignableFrom(actualClass));
046          return actualClass;
047        } catch (ClassNotFoundException e) {
048          throw new IllegalArgumentException("Could not find class " + raw);
049        }
050      }
051    }
