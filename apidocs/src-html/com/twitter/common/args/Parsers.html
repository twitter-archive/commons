017    package com.twitter.common.args;
018    
019    import java.util.Map;
020    
021    import com.google.common.base.Function;
022    import com.google.common.base.Preconditions;
023    import com.google.common.base.Splitter;
024    import com.google.common.collect.ImmutableMap;
025    import com.google.common.collect.Maps;
026    
027    import com.twitter.common.args.apt.Configuration;
028    import com.twitter.common.args.apt.Configuration.ParserInfo;
029    
030    import static com.google.common.base.Preconditions.checkArgument;
031    import static com.twitter.common.args.apt.Configuration.ConfigurationException;
032    
033    /**
034     * A registry of Parsers for different supported argument types.
035     *
036     * @author William Farner
037     */
038    public final class Parsers implements ParserOracle {
039    
040      private final ImmutableMap<Class<?>, Parser<?>> registry;
041    
042      /**
043       * Creates a new parser registry over the specified {@code parsers}.
044       *
045       * @param parsers The parsers to register.
046       */
047      public Parsers(Map<Class<?>, Parser<?>> parsers) {
048        Preconditions.checkNotNull(parsers);
049        registry = ImmutableMap.copyOf(parsers);
050      }
051    
052      @Override
053      public Parser get(Class<?> cls) throws IllegalArgumentException {
054        Parser parser;
055        Class<?> explicitClass = cls;
056        while (((parser = registry.get(explicitClass)) == null) && (explicitClass != null)) {
057          explicitClass = explicitClass.getSuperclass();
058        }
059        checkArgument(parser != null, "No parser found for " + cls);
060        return parser;
061      }
062    
063      private static final Function<ParserInfo, Class<?>> INFO_TO_PARSED_TYPE =
064          new Function<ParserInfo, Class<?>>() {
065            @Override public Class<?> apply(ParserInfo parserInfo) {
066              try {
067                return Class.forName(parserInfo.parsedType);
068              } catch (ClassNotFoundException e) {
069                throw new ConfigurationException(e);
070              }
071            }
072          };
073    
074      private static final Function<ParserInfo, Parser<?>> INFO_TO_PARSER =
075          new Function<ParserInfo, Parser<?>>() {
076            @Override public Parser apply(ParserInfo parserInfo) {
077              try {
078                return (Parser<?>) Class.forName(parserInfo.parserClass).newInstance();
079              } catch (ClassNotFoundException e) {
080                throw new ConfigurationException(e);
081              } catch (InstantiationException e) {
082                throw new ConfigurationException(e);
083              } catch (IllegalAccessException e) {
084                throw new ConfigurationException(e);
085              }
086            }
087          };
088    
089      static Parsers fromConfiguration(Configuration configuration) {
090        Map<Class<?>, Parser<?>> parsers =
091            Maps.transformValues(
092                Maps.uniqueIndex(configuration.parserInfo(), INFO_TO_PARSED_TYPE),
093                INFO_TO_PARSER);
094        return new Parsers(parsers);
095      }
096    
097      public static final Splitter MULTI_VALUE_SPLITTER =
098            Splitter.on(",").trimResults().omitEmptyStrings();
099    }
