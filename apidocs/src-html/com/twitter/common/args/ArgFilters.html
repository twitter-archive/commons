017    package com.twitter.common.args;
018    
019    import java.lang.reflect.Field;
020    import java.util.Set;
021    
022    import com.google.common.base.Preconditions;
023    import com.google.common.base.Predicate;
024    import com.google.common.base.Predicates;
025    import com.google.common.collect.ImmutableSet;
026    
027    import com.twitter.common.base.MorePreconditions;
028    
029    /**
030     * Utilities for generating {@literal @CmdLine} {@link Arg} filters suitable for use with
031     * {@link com.twitter.common.args.ArgScanner#parse(Predicate, Iterable)}.  These filters assume the
032     * fields parsed will all be annotated with {@link CmdLine}.
033     *
034     * @author John Sirois
035     */
036    public final class ArgFilters {
037      private ArgFilters() {
038        // utility
039      }
040    
041      /**
042       * A filter that selects all {@literal @CmdLine} {@link Arg}s found on the classpath.
043       */
044      public static final Predicate<Field> SELECT_ALL = Predicates.alwaysTrue();
045    
046      /**
047       * Creates a filter that selects all {@literal @CmdLine} {@link Arg}s found in classes that are
048       * members of the given package.  Note that this will not select subpackages.
049       *
050       * @param pkg The exact package of classes whose command line args will be selected.
051       * @return A filter that selects only command line args declared in classes that are members of
052       *     the given {@code pkg}.
053       */
054      public static Predicate<Field> selectPackage(final Package pkg) {
055        Preconditions.checkNotNull(pkg);
056        return new Predicate<Field>() {
057          @Override public boolean apply(Field field) {
058            return field.getDeclaringClass().getPackage().equals(pkg);
059          }
060        };
061      }
062    
063      /**
064       * Creates a filter that selects all {@literal @CmdLine} {@link Arg}s found in classes that are
065       * members of the given package or its sub-packages.
066       *
067       * @param pkg The ancestor package of classes whose command line args will be selected.
068       * @return A filter that selects only command line args declared in classes that are members of
069       *     the given {@code pkg} or its sub-packages.
070       */
071      public static Predicate<Field> selectAllPackagesUnderHere(final Package pkg) {
072        Preconditions.checkNotNull(pkg);
073        final String prefix = pkg.getName() + '.';
074        return Predicates.or(selectPackage(pkg), new Predicate<Field>() {
075          @Override public boolean apply(Field field) {
076            return field.getDeclaringClass().getPackage().getName().startsWith(prefix);
077          }
078        });
079      }
080    
081      /**
082       * Creates a filter that selects all {@literal @CmdLine} {@link Arg}s found in the given class.
083       *
084       * @param clazz The class whose command line args will be selected.
085       * @return A filter that selects only command line args declared in the given {@code clazz}.
086       */
087      public static Predicate<Field> selectClass(final Class<?> clazz) {
088        Preconditions.checkNotNull(clazz);
089        return new Predicate<Field>() {
090          @Override public boolean apply(Field field) {
091            return field.getDeclaringClass().equals(clazz);
092          }
093        };
094      }
095    
096      /**
097       * Creates a filter that selects all {@literal @CmdLine} {@link Arg}s found in the given classes.
098       *
099       * @param cls The classes whose command line args will be selected.
100       * @return A filter that selects only command line args declared in the given classes.
101       */
102      public static Predicate<Field> selectClasses(final Class<?> ... cls) {
103        Preconditions.checkNotNull(cls);
104        final Set<Class<?>> listOfClasses = ImmutableSet.copyOf(cls);
105        return new Predicate<Field>() {
106          @Override public boolean apply(Field field) {
107            return listOfClasses.contains(field.getDeclaringClass());
108          }
109        };
110      }
111    
112      /**
113       * Creates a filter that selects a single {@literal @CmdLine} {@link Arg}.
114       *
115       * @param clazz The class that declares the command line arg to be selected.
116       * @param name The {@link com.twitter.common.args.CmdLine#name()} of the arg to select.
117       * @return A filter that selects a single specified command line arg.
118       */
119      public static Predicate<Field> selectCmdLineArg(Class<?> clazz, final String name) {
120        MorePreconditions.checkNotBlank(name);
121        return Predicates.and(selectClass(clazz), new Predicate<Field>() {
122          @Override public boolean apply(Field field) {
123            return field.getAnnotation(CmdLine.class).name().equals(name);
124          }
125        });
126      }
127    }
