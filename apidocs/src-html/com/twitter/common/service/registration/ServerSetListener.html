018    package com.twitter.common.service.registration;
019    
020    import java.util.Set;
021    
022    /**
023     * An interface to an object that is interested in receiving notification whenever the
024     * membership of a {@link ServerSet} changes. See {@link ServerSet.setListener()}.
025     * @author Patrick Chan
026     */
027    public interface ServerSetListener {
028      /**
029       * After the listener is registered, the first call will contain the current server
030       * set membership. The method is then called every time the membership changes.
031       * The returned set is a copy of the server set's internal data. The initial
032       * membership of a server set is empty. When disconnected, the membership won't change.
033       *
034       * The implementation of this method should return in a timely manner.
035       * This method could be called more than once with the same value.
036       * This method will not be called concurrently.
037       *
038       * @param serverSet  the non-null copy of the current set of available servers.
039       */
040      void onChange(Set<Server> serverSet);
041    
042      /**
043       * Although a server set tries to mask connection failures, a server could become
044       * disconnected from the other members. This method is called whenever there's a
045       * change in the connection status of the server set. The initial state of a server
046       * set is disconnected.
047       *
048       * The implementation of this method should return in a timely manner.
049       * This method could be called more than once with the same value.
050       * This method will not be called concurrently.
051       */
052      void onConnect(boolean connected);
053    }
