046    /**
047     * This ServerSet implementation uses ZooKeeper to implement membership.
048     * @author Patrick Chan
049     */
050    public class ZkServerSet implements ServerSet {
051      private static final Logger LOG = Logger.getLogger(ZkServerSet.class.getName());
052      private static final int MAX_SOCKET_NUMBER = 0xFFFF;
053      private static final int ZK_TIMEOUT = 10;
054    
055      private ZooKeeperClient zkClient;
056    
057      // Input
058      private final String host;
059      private final int port;
060      private final String path;
061    
062      // Background threads
063      private ServerSetListener listener;
064      private JoinThread joinThread;
065      private ListenerThread listenerThread;
066      private final BackoffHelper backoffHelper = new BackoffHelper();
067    
068      // Current state
069      private Map<String, Server> servers = new HashMap<String, Server>();
070      private boolean isConnected = false;
071    
072      // ZooKeeper
073      private Group group;
074      private Group.Membership membership;
075      private com.twitter.common.zookeeper.ServerSet sset;

077      /**
078       * Constructs a server set based on the supplied parameters.
079       * The returned object can be used to immediately join the server set
080       * (see {@link join()} and/or to discover the existing members in the server set
081       * (see {@link setListener()}).
082       *
083       * The supplied names must consist only of alphanumerics; no punctuation or white
084       * space is allowed.
085       *
086       * This constructor tries to catch and handle all connection-related exceptions.
087       * Any runtime exception that is thrown indicates a permanent configuration or programming error.
088       *
089       * @param host        non-null host name or IP for a zookeeper server
090       * @param port        the port of the zookeeper server
091       * @param path        non-null zookeeper path
092       */
093      public ZkServerSet(String host, int port, String path) {
094        Preconditions.checkNotNull(host);
095        Preconditions.checkArgument(port > 0 && port < MAX_SOCKET_NUMBER,
096                                    "port must be > 0 and < 0xFFFF");
097        Preconditions.checkNotNull(path);
098    
099        this.host = host;
100        this.port = port;
101        this.path = path;
102    
103        // If the server doesn't hear from the client by the timeout
104        // any empheral nodes created by the client will be removed.
105        zkClient = new ZooKeeperClient(Amount.of(ZK_TIMEOUT, Time.SECONDS),
106            ImmutableList.of(InetSocketAddress.createUnresolved(host, port)));
107        zkClient.register(new MyWatcher());
108        group = new Group(zkClient, ZooDefs.Ids.OPEN_ACL_UNSAFE, path, "server_");
109        sset = new com.twitter.common.zookeeper.ServerSetImpl(zkClient, group);
110      }
111    
112      /**
113       * Watcher for zookeeper events.
114       */
115      class MyWatcher implements Watcher {
116        @Override public void process(WatchedEvent event) {
117          if (LOG.isLoggable(Level.FINE)) {
118            LOG.log(Level.FINE, "" + event);
119          }
120          if (event.getType() == Watcher.Event.EventType.None) {
121            switch (event.getState()) {
122              case Disconnected:
123                if (listener != null) {
124                  listener.onConnect(false);
125                }
126                break;
127              case SyncConnected:
128                if (listener != null) {
129                  listener.onConnect(true);
130                }
131                break;
132              case Expired:         // Do nothing since zkClient will reestablish connection
133              case NoSyncConnected: // Deprecated
134              case Unknown:
135              default:
136                break;
137            }
138          }
139        }
140      };

143      /**
144       * See {@link ServerSet.join()} for details.
145       */
146      public void join(Server server) {
147        if (joinThread != null) {
148          throw new IllegalStateException("join() cannot be called again until unjoin() is called.");
149        }
150        joinThread = new JoinThread(server);
151        joinThread.setDaemon(true);
152        joinThread.start();
153      }
154    
155      /**
156       * This thread keeps trying to join the group until the join is successful.
157       * Once joined, it waits around until unjoin() is called.
158       * Note: it might be better to have the thread exit after the join succeeds.
159       * Just need to handle the race condition of an unjoin() call just as the
160       * join succeeds. Need to make sure membership.cancel() is called if the
161       * membership object is successfully created.
162       */
163      class JoinThread extends Thread {
164        private Server server;
165        private boolean cancelled = false;
166    
167        JoinThread(Server server) {
168          this.server = server;
169        }
170    
171        public synchronized void cancel() {
172          cancelled = true;
173          notify();
174        }
175    
176        public void run() {
177          try {
178            // Prepare the serialized data
179            final Supplier<byte[]> data = new Supplier<byte[]>() {
180              @Override public byte[] get() {
181                return server.getSerialized();
182              }
183            };
184    
185            // Continue call join() until it succeeds
186            backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, RuntimeException>() {
187              @Override public Boolean get() {
188                try {
189                  if (!cancelled) {
190                    membership = group.join(data);
191                  }
192                  return true;
193                } catch (Exception e) {
194                  LOG.log(Level.WARNING, "Join thread failed to join as " + server
195                          + ". Will retry.", e);
196                }
197                return false;
198              }
199            });
200    
201            // Join succeeded. Now wait for cancelled.
202            synchronized (this) {
203              while (!cancelled) {
204                wait();
205              }
206            }
207    
208            // unjoin() was called so cancel the membership
209            // TODO: If an exception occurs, do we need to try again?
210            if (membership != null) {
211              try {
212                membership.cancel();
213              } catch (JoinException e) {
214                LOG.log(Level.WARNING, "Join thread cancel() failed for '" + server + "'. Ignored.", e);
215              }
216            }
217          } catch (InterruptedException e) {
218            LOG.log(Level.WARNING, "Join thread was interrupted, which should never happen.", e);
219          }
220        }
221      }
222    
223      /**
224       * See {@link ServerSet.unjoin()} for details.
225       */
226      public void unjoin(Server server) {
227        if (joinThread == null) {
228          throw new IllegalStateException("join() has not yet been called.");
229        }
230        if (!joinThread.server.equals(server)) {
231          throw new IllegalStateException(
232            "The supplied server object does not equal to the object supplied to join().");
233        }
234        if (joinThread != null) {
235          joinThread.cancel();
236          joinThread = null;
237        }
238      }

240      /**
241       * See {@link ServerSet.setListener()} for details.
242       */
243      public void setListener(ServerSetListener listener) {
244        if (this.listener != null) {
245          throw new IllegalStateException("setListener() cannot be called again.");
246        }
247        this.listener = listener;
248    
249        // Initialize connection state.
250        // There's a race condition that exists between the watcher and this thread
251        // which can cause the listener to be left with incorrect state.
252        // E.g. this thread sees that isConnected is false and will invoke
253        // the callback with false. Before the callback is called, the watcher sets
254        // isConnected to true. When this thread finally gets around to invoking the callback,
255        // it does so with old state.
256        // To fix the race condition, a check is made before and after the callback.
257        // If it's different, the callback is invoked again.
258        // Although the callback might be invoked with the same value, the state
259        // will eventually be consistent.
260        boolean b = false;
261        do {
262          b = isConnected;
263          listener.onConnect(b);
264        } while (b != isConnected);
265    
266        listenerThread = new ListenerThread();
267        listenerThread.setDaemon(true);
268        listenerThread.start();
269      }
270    
271      /**
272       * This thread continues to try and set a watch. Once the watch is set, this thread
273       * terminates.
274       */
275      class ListenerThread extends Thread {
276        public void run() {
277          try {
278            // Set a watch to look for group changes.
279            // Continues to call watch() until it succeeds.
280            backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, RuntimeException>() {
281              @Override public Boolean get() {
282                try {
283                  // This method blocks if no connectivity to ZK.
284                  // The watch will invoke the callback with the initial state.
285                  group.watch(new GroupChangeListener() {
286                    @Override public void onGroupChange(Iterable<String> memberIds) {
287                      notifyGroupChange(memberIds);
288                    }
289                  });
290                  return true;
291                } catch (Exception e) {
292                  LOG.log(Level.WARNING, "Join thread failed to set watch. Will retry.", e);
293                }
294                return false;
295              }
296            });
297          } catch (InterruptedException e) {
298            LOG.log(Level.WARNING, "Listener thread was interrupted, which should never happen.", e);
299          }
300        }
301      }
302    
303      /**
304       * This method takes the list of live znode names and updates the server set's list of servers
305       * and invokes the callback if there are any changes.
306       * Any errors encountered while trying to retrieve node information for a server is
307       * logged and ignored and the server is not included in the set.
308       *
309       * This method is thread-safe.
310       *
311       * @param memberIds non-null list of the latest live empheral znode names.
312       */
313      void notifyGroupChange(Iterable<String> memberIds) {
314        // Do all the calculations on a copy
315        Map<String, Server> serversCopy = new HashMap<String, Server>(servers);
316    
317        // Check if there are any changes
318        Set<String> latest = new HashSet<String>();
319        for (String m : memberIds) {
320          latest.add(m);
321        }
322        if (latest.equals(serversCopy.keySet())) {
323          return;
324        }
325    
326        // Add new servers
327        for (String m : memberIds) {
328          if (!serversCopy.containsKey(m)) {
329            try {
330              // Get the data
331              Server server = Server.deserialize(group.getMemberData(m));
332              serversCopy.put(m, server);
333            } catch (Exception e) {
334              LOG.log(Level.WARNING, "Could not deserialize data for server " + m, e);
335            }
336          }
337        }
338    
339        // Remove missing servers
340        Iterator it = serversCopy.keySet().iterator();
341        while (it.hasNext()) {
342          if (!latest.contains(it.next())) {
343            it.remove();
344          }
345        }
346    
347        // Call the listener with a copy of the server set
348        listener.onChange(new HashSet<Server>(serversCopy.values()));
349    
350        // Update the current list of servers with the latest information
351        servers = serversCopy;
352      }

354      ZooKeeperClient getZkClient() {
355        return zkClient;
356      }
