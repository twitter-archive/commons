001    package com.twitter.common.logging;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    import java.util.logging.ConsoleHandler;
006    import java.util.logging.Handler;
007    import java.util.logging.Level;
008    import java.util.logging.LogManager;
009    import java.util.logging.Logger;
010    
011    import com.google.common.annotations.VisibleForTesting;
012    import com.google.common.base.Preconditions;
013    
014    import com.twitter.common.args.Arg;
015    import com.twitter.common.args.CmdLine;
016    
017    /**
018     * A configuration class for the root java.util.logging Logger.
019     *
020     * Defines flags to control the behavior behavior of the root logger similarly to Google's glog
021     * library (see http://code.google.com/p/google-glog ).
022     */
023    public class RootLogConfig {
024      /**
025       * An enum reflecting log {@link Level} constants.
026       */
027      public enum LogLevel {
028        FINEST(Level.FINEST),
029        FINER(Level.FINER),
030        FINE(Level.FINE),
031        CONFIG(Level.CONFIG),
032        INFO(Level.INFO),
033        WARNING(Level.WARNING),
034        SEVERE(Level.SEVERE);
035    
036        private final Level level;
037    
038        private LogLevel(Level level) {
039          this.level = level;
040        }
041    
042        private Level getLevel() {
043          return level;
044        }
045    
046        private int intValue() {
047          return level.intValue();
048        }
049      }
050    
051      @CmdLine(name = "logtostderr", help = "Log messages to stderr instead of logfiles.")
052      private static Arg<Boolean> LOGTOSTDERR = Arg.create(false);
053    
054      @CmdLine(name = "alsologtostderr",
055               help = "Log messages to stderr, in addition to log files. Ignored when --logtostderr")
056      private static Arg<Boolean> ALSOLOGTOSTDERR = Arg.create(false);
057    
058      @CmdLine(name = "vlog",
059               help = "The value is one of the constants in java.util.logging.Level. "
060                      + "Shows all messages with level equal or higher "
061                      + "than the value of this flag.")
062      private static Arg<LogLevel> VLOG = Arg.create(LogLevel.INFO);
063    
064      @CmdLine(name = "vmodule",
065               help = "Per-class verbose level. The argument has to contain a comma-separated list "
066                      + "of <class_name>=<log_level>. <class_name> is the full-qualified name of a "
067                      + "class, <log_level> is one of the constants in java.util.logging.Level. "
068                      + "<log_level> overrides any value given by --vlog.")
069      private static Arg<Map<Class<?>, LogLevel>> VMODULE =
070          Arg.<Map<Class<?>, LogLevel>>create(new HashMap<Class<?>, LogLevel>());
071    
072      // TODO(franco): change this flag's default to true, then remove after enough forewarning.
073      @CmdLine(name = "use_glog_formatter", help = "True to use the glog formatter exclusively.")
074      private static Arg<Boolean> USE_GLOG_FORMATTER = Arg.create(false);
075    
076      /**
077       * A builder-pattern class used to perform the configuration programmatically
078       * (i.e. not through flags).
079       * Example:
080       * <code>
081       *    RootLogConfig.builder().logToStderr(true).apply();
082       * </code>
083       */
084      public static class Configuration {
085        private boolean logToStderr = false;
086        private boolean alsoLogToStderr = false;
087        private boolean useGLogFormatter = false;
088        private LogLevel vlog = null;
089        private Map<Class<?>, LogLevel> vmodule = null;
090        private String rootLoggerName = "";
091    
092        private Configuration() {}
093        /**
094         * Only log messages to stderr, instead of log files. Overrides alsologtostderr.
095         * Default: false.
096         *
097         * @param flag True to enable, false to disable.
098         * @return this Configuration object.
099         */
100        public Configuration logToStderr(boolean flag) {
101          this.logToStderr = flag;
102          return this;
103        }
104    
105        /**
106         * Also log messages to stderr, in addition to log files.
107         * Overridden by logtostderr.
108         * Default: false.
109         *
110         * @param flag True to enable, false to disable.
111         * @return this Configuration object.
112         */
113        public Configuration alsoLogToStderr(boolean flag) {
114          this.alsoLogToStderr = flag;
115          return this;
116        }
117    
118        /**
119         * Format log messages in one-line with a header, similar to google-glog.
120         * Default: false.
121         *
122         * @param flag True to enable, false to disable.
123         * @return this Configuration object.
124         */
125        public Configuration useGLogFormatter(boolean flag) {
126          this.useGLogFormatter = flag;
127          return this;
128        }
129    
130        /**
131         * Output log messages at least at the given verbosity level.
132         * Overridden by vmodule.
133         * Default: INFO
134         *
135         * @param level LogLevel enumerator for the minimum log message verbosity level that is output.
136         * @return this Configuration object.
137         */
138        public Configuration vlog(LogLevel level) {
139          Preconditions.checkNotNull(level);
140          this.vlog = level;
141          return this;
142        }
143    
144        /**
145         * Output log messages for a given set of classes at the associated verbosity levels.
146         * Overrides vlog.
147         * Default: no classes are treated specially.
148         *
149         * @param pairs Map of classes and correspoding log levels.
150         * @return this Configuration object.
151         */
152        public Configuration vmodule(Map<Class<?>, LogLevel> pairs) {
153          Preconditions.checkNotNull(pairs);
154          this.vmodule = pairs;
155          return this;
156        }
157    
158        /**
159         * Applies this configuration to the root log.
160         */
161        public void apply() {
162          RootLogConfig.configure(this);
163        }
164    
165        // Intercepts the root logger, for testing purposes only.
166        @VisibleForTesting
167        Configuration rootLoggerName(String name) {
168          Preconditions.checkNotNull(name);
169          Preconditions.checkArgument(!name.isEmpty());
170          this.rootLoggerName = name;
171          return this;
172        }
173      }
174    
175      /**
176       * Creates a new Configuration builder object.
177       * @return The newly built Configuration.
178       */
179      public static Configuration builder() {
180        return new Configuration();
181      }
182    
183      /**
184       *  Configures the root log properties using flags.
185       *  This is the entry point used by AbstractApplication via LogModule.
186       */
187      public static void configureFromFlags() {
188        builder()
189            .logToStderr(LOGTOSTDERR.get())
190            .alsoLogToStderr(ALSOLOGTOSTDERR.get())
191            .useGLogFormatter(USE_GLOG_FORMATTER.get())
192            .vlog(VLOG.get())
193            .vmodule(VMODULE.get())
194            .apply();
195      }
196    
197      private static void configure(Configuration configuration) {
198        // Edit the properties of the root logger.
199        Logger rootLogger = Logger.getLogger(configuration.rootLoggerName);
200        if (configuration.logToStderr) {
201          setLoggerToStderr(rootLogger);
202        } else if (configuration.alsoLogToStderr) {
203          setLoggerToAlsoStderr(rootLogger);
204        }
205        if (configuration.useGLogFormatter) {
206          setGLogFormatter(rootLogger);
207        }
208        if (configuration.vlog != null) {
209          setVlog(rootLogger, configuration.vlog);
210        }
211        if (configuration.vmodule != null) {
212          setVmodules(configuration.vmodule);
213        }
214      }
215    
216      private static void setLoggerToStderr(Logger logger) {
217        LogManager.getLogManager().reset();
218        setConsoleHandler(logger, true);
219      }
220    
221      private static void setLoggerToAlsoStderr(Logger logger) {
222        setConsoleHandler(logger, false);
223      }
224    
225      private static void setConsoleHandler(Logger logger, boolean removeOtherHandlers) {
226        Handler consoleHandler = null;
227        for (Handler h : logger.getHandlers()) {
228          if (h instanceof ConsoleHandler) {
229            consoleHandler = h;
230          } else if (removeOtherHandlers) {
231            logger.removeHandler(h);
232          }
233        }
234        if (consoleHandler == null) {
235          consoleHandler = new ConsoleHandler();
236          logger.addHandler(new ConsoleHandler());
237        }
238        consoleHandler.setLevel(Level.ALL);
239        consoleHandler.setFilter(null);
240      }
241    
242      private static void setGLogFormatter(Logger logger) {
243        for (Handler h : logger.getHandlers()) {
244          h.setFormatter(new LogFormatter());
245        }
246      }
247    
248      private static void setVmodules(Map<Class<?>, LogLevel> vmodules) {
249        for (Map.Entry<Class<?>, LogLevel> entry : vmodules.entrySet()) {
250          String className = entry.getKey().getName();
251          Logger logger = Logger.getLogger(className);
252          setVlog(logger, entry.getValue());
253        }
254      }
255    
256      private static void setVlog(Logger logger, LogLevel logLevel) {
257        final Level newLevel = logLevel.getLevel();
258        logger.setLevel(newLevel);
259        do {
260          for (Handler handler : logger.getHandlers()) {
261            Level handlerLevel = handler.getLevel();
262            if (newLevel.intValue() < handlerLevel.intValue()) {
263              handler.setLevel(newLevel);
264            }
265          }
266        } while (logger.getUseParentHandlers() && (logger = logger.getParent()) != null);
267      }
268    
269      // Utility class.
270      private RootLogConfig() {
271      }
272    }
