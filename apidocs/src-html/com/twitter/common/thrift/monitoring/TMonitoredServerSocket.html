017    package com.twitter.common.thrift.monitoring;
018    
019    import com.google.common.base.Preconditions;
020    import com.google.common.collect.Maps;
021    import com.twitter.common.net.monitoring.ConnectionMonitor;
022    import org.apache.thrift.transport.TServerSocket;
023    import org.apache.thrift.transport.TSocket;
024    import org.apache.thrift.transport.TTransportException;
025    
026    import java.net.InetSocketAddress;
027    import java.net.ServerSocket;
028    import java.util.Collections;
029    import java.util.Map;
030    
031    /**
032     * Extension of TServerSocket that allows for tracking of connected clients.
033     *
034     * @author William Farner
035     */
036    public class TMonitoredServerSocket extends TServerSocket {
037      private ConnectionMonitor<InetSocketAddress> monitor;
038    
039      public TMonitoredServerSocket(ServerSocket serverSocket,
040          ConnectionMonitor<InetSocketAddress> monitor) {
041        super(serverSocket);
042        this.monitor = Preconditions.checkNotNull(monitor);
043      }
044    
045      public TMonitoredServerSocket(ServerSocket serverSocket, int clientTimeout,
046          ConnectionMonitor<InetSocketAddress> monitor) {
047        super(serverSocket, clientTimeout);
048        this.monitor = Preconditions.checkNotNull(monitor);
049      }
050    
051      public TMonitoredServerSocket(int port, ConnectionMonitor<InetSocketAddress> monitor)
052          throws TTransportException {
053        super(port);
054        this.monitor = Preconditions.checkNotNull(monitor);
055      }
056    
057      public TMonitoredServerSocket(int port, int clientTimeout,
058          ConnectionMonitor<InetSocketAddress> monitor) throws TTransportException {
059        super(port, clientTimeout);
060        this.monitor = Preconditions.checkNotNull(monitor);
061      }
062    
063      public TMonitoredServerSocket(InetSocketAddress bindAddr,
064          ConnectionMonitor<InetSocketAddress> monitor) throws TTransportException {
065        super(bindAddr);
066        this.monitor = Preconditions.checkNotNull(monitor);
067      }
068    
069      public TMonitoredServerSocket(InetSocketAddress bindAddr, int clientTimeout,
070          ConnectionMonitor<InetSocketAddress> monitor) throws TTransportException {
071        super(bindAddr, clientTimeout);
072        this.monitor = Preconditions.checkNotNull(monitor);
073      }
074    
075      private final Map<TSocket, InetSocketAddress> addressMap =
076          Collections.synchronizedMap(Maps.<TSocket, InetSocketAddress>newHashMap());
077    
078      public InetSocketAddress getAddress(TSocket socket) {
079        return addressMap.get(socket);
080      }
081    
082      @Override
083      protected TSocket acceptImpl() throws TTransportException {
084        final TSocket socket = super.acceptImpl();
085        final InetSocketAddress remoteAddress =
086            (InetSocketAddress) socket.getSocket().getRemoteSocketAddress();
087    
088        TSocket monitoredSocket = new TSocket(socket.getSocket()) {
089          boolean closed = false;
090    
091          @Override public void close() {
092            try {
093              super.close();
094            } finally {
095              if (!closed) {
096                monitor.released(remoteAddress);
097                addressMap.remove(this);
098              }
099              closed = true;
100            }
101          }
102        };
103    
104        addressMap.put(monitoredSocket, remoteAddress);
105    
106        monitor.connected(remoteAddress);
107        return monitoredSocket;
108      }
109    
110      @Override
111      public void close() {
112        super.close();
113      }
114    }
