017    package com.twitter.common.util;
018    
019    import com.google.common.base.Preconditions;
020    import com.twitter.common.quantity.Amount;
021    import com.twitter.common.quantity.Time;
022    
023    /**
024     * A BackoffStrategy that implements truncated binary exponential backoff.
025     */
026    public class TruncatedBinaryBackoff implements BackoffStrategy {
027      private final long initialBackoffMs;
028      private final long maxBackoffIntervalMs;
029      private final boolean stopAtMax;
030    
031      private volatile boolean stop = false;
032    
033      /**
034       * Creates a new TruncatedBinaryBackoff that will start by backing off for {@code initialBackoff}
035       * and then backoff of twice as long each time its called until reaching the {@code maxBackoff} at
036       * which point shouldContinue() will return false and any future backoffs will always wait for
037       * that amount of time.
038       *
039       * @param initialBackoff the intial amount of time to backoff
040       * @param maxBackoff the maximum amount of time to backoff
041       * @param stopAtMax whether shouldContinue() returns false when the max is reached
042       */
043      public TruncatedBinaryBackoff(Amount<Long, Time> initialBackoff,
044          Amount<Long, Time> maxBackoff, boolean stopAtMax) {
045        Preconditions.checkNotNull(initialBackoff);
046        Preconditions.checkNotNull(maxBackoff);
047        Preconditions.checkArgument(initialBackoff.getValue() > 0);
048        Preconditions.checkArgument(maxBackoff.compareTo(initialBackoff) >= 0);
049        initialBackoffMs = initialBackoff.as(Time.MILLISECONDS);
050        maxBackoffIntervalMs = maxBackoff.as(Time.MILLISECONDS);
051        this.stopAtMax = stopAtMax;
052      }
053    
054      /**
055       * Same as main constructor, but this will always return true from shouldContinue().
056       *
057       * @param initialBackoff the intial amount of time to backoff
058       * @param maxBackoff the maximum amount of time to backoff
059       */
060      public TruncatedBinaryBackoff(Amount<Long, Time> initialBackoff, Amount<Long, Time> maxBackoff) {
061        this(initialBackoff, maxBackoff, false);
062      }
063    
064      @Override
065      public long calculateBackoffMs(long lastBackoffMs) {
066        Preconditions.checkArgument(lastBackoffMs >= 0);
067        long backoff = (lastBackoffMs == 0) ? initialBackoffMs
068            : Math.min(maxBackoffIntervalMs, lastBackoffMs * 2);
069        stop = stop || (stopAtMax && (backoff >= maxBackoffIntervalMs));
070        return backoff;
071      }
072    
073      @Override
074      public boolean shouldContinue() {
075        return !stop;
076      }
077    }
