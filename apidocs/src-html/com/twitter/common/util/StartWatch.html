028    package com.twitter.common.util;
029    
030    import org.apache.commons.lang.time.StopWatch;
031    
032    public class StartWatch extends StopWatch {
033      public StartWatch() {
034        super();
035      }
036    
037      public void start() {
038        _started = true;
039        super.start();
040      }
041    
042      public void resume() {
043        if (!_started) {
044          start();
045        } else {
046          super.resume();
047        }
048      }
049    
050      private boolean _started = false;
051    }
