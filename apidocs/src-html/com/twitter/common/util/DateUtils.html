017    package com.twitter.common.util;
018    
019    import java.util.Calendar;
020    import java.util.Date;
021    import java.util.concurrent.TimeUnit;
022    
023    /**
024     * Utilities for working with java {@link Date}s.
025     *
026     * @author John Sirois
027     */
028    public final class DateUtils {
029    
030      public static Date now() {
031        return new Date();
032      }
033    
034      public static long toUnixTime(Date date) {
035        return toUnixTime(date.getTime());
036      }
037    
038      public static long nowUnixTime() {
039        return toUnixTime(System.currentTimeMillis());
040      }
041    
042      public static long toUnixTime(long millisSinceEpoch) {
043        return TimeUnit.MILLISECONDS.toSeconds(millisSinceEpoch);
044      }
045    
046      public static Date ago(int calendarField, int amount) {
047        return ago(now(), calendarField, amount);
048      }
049    
050       public static Date ago(Date referenceDate, int calendarField, int amount) {
051        Calendar calendar = Calendar.getInstance();
052        calendar.setTime(referenceDate);
053        calendar.add(calendarField, -1 * amount);
054        return calendar.getTime();
055      }
056    
057      private DateUtils() {
058        // utility
059      }
060    }
