017    package com.twitter.common.util.concurrent;
018    
019    import com.google.common.base.Preconditions;
020    
021    import java.util.concurrent.Callable;
022    import java.util.concurrent.ExecutorService;
023    import java.util.concurrent.FutureTask;
024    import java.util.logging.Level;
025    import java.util.logging.Logger;
026    
027    /**
028     * A future task that supports retries by resubmitting itself to an {@link ExecutorService}.
029     *
030     * @author William Farner
031     */
032    public class RetryingFutureTask extends FutureTask<Boolean> {
033      private static Logger LOG = Logger.getLogger(RetryingFutureTask.class.getName());
034    
035      protected final ExecutorService executor;
036      protected final int maxRetries;
037      protected int numRetries = 0;
038      protected final Callable<Boolean> callable;
039    
040      /**
041       * Creates a new retrying future task that will execute a unit of work until successfully
042       * completed, or the retry limit has been reached.
043       *
044       * @param executor The executor service to resubmit the task to upon failure.
045       * @param callable The unit of work.  The work is considered successful when {@code true} is
046       *    returned.  It may return {@code false} or throw an exception when unsueccessful.
047       * @param maxRetries The maximum number of times to retry the task.
048       */
049      public RetryingFutureTask(ExecutorService executor, Callable<Boolean> callable, int maxRetries) {
050        super(callable);
051        this.callable = Preconditions.checkNotNull(callable);
052        this.executor = Preconditions.checkNotNull(executor);
053        this.maxRetries = maxRetries;
054      }
055    
056      /**
057       * Invokes a retry of this task.
058       */
059      protected void retry() {
060        executor.execute(this);
061      }
062    
063      @Override
064      public void run() {
065        boolean success = false;
066        try {
067          success = callable.call();
068        } catch (Exception e) {
069          LOG.log(Level.WARNING, "Exception while executing task.", e);
070        }
071    
072        if (!success) {
073          numRetries++;
074          if (numRetries > maxRetries) {
075            LOG.severe("Task did not complete after " + maxRetries + " retries, giving up.");
076          } else {
077            LOG.info("Task was not successful, resubmitting (num retries: " + numRetries + ")");
078            retry();
079          }
080        } else {
081          set(true);
082        }
083      }
084    }
