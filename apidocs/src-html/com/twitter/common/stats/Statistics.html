018    package com.twitter.common.stats;
019    
020    /**
021     * A simple class to keep running statistics that require O(1) storage.
022     *
023     * @author William Farner
024     */
025    public class Statistics {
026      private long populationSize = 0;
027      private double accumulatedVariance = 0;
028      private double runningMean = 0;
029    
030      private long minValue = Long.MAX_VALUE;
031      private long maxValue = Long.MIN_VALUE;
032    
033      public void accumulate(long value) {
034        populationSize++;
035        double delta = value - runningMean;
036        runningMean += delta / populationSize;
037        accumulatedVariance += delta * (value - runningMean);
038    
039        // Update max/min.
040        minValue = value < minValue ? value : minValue;
041        maxValue = value > maxValue ? value : maxValue;
042      }
043    
044      public double variance() {
045        return accumulatedVariance / populationSize;
046      }
047    
048      public double standardDeviation() {
049        return Math.sqrt(variance());
050      }
051    
052      public double mean() {
053        return runningMean;
054      }
055    
056      public long min() {
057        return minValue;
058      }
059    
060      public long max() {
061        return maxValue;
062      }
063    
064      public long range() {
065        return maxValue - minValue;
066      }
067    
068      public long populationSize() {
069        return populationSize;
070      }
071    
072      @Override
073      public String toString() {
074        return String.format("Mean: %f, Min: %d, Max: %d, Range: %d, Stddev: %f, Variance: %f",
075            mean(), min(), max(), range(), standardDeviation(), variance());
076      }
077    }
