017    package com.twitter.common.stats;
018    
019    /**
020     * A simple class to keep running statistics that require O(1) storage.
021     *
022     * @author William Farner
023     */
024    public class Statistics {
025      private long populationSize = 0;
026      private double accumulatedVariance = 0;
027      private double runningMean = 0;
028    
029      private long minValue = Long.MAX_VALUE;
030      private long maxValue = Long.MIN_VALUE;
031    
032      public void accumulate(long value) {
033        populationSize++;
034        double delta = value - runningMean;
035        runningMean += delta / populationSize;
036        accumulatedVariance += delta * (value - runningMean);
037    
038        // Update max/min.
039        minValue = value < minValue ? value : minValue;
040        maxValue = value > maxValue ? value : maxValue;
041      }
042    
043      public double variance() {
044        return accumulatedVariance / populationSize;
045      }
046    
047      public double standardDeviation() {
048        return Math.sqrt(variance());
049      }
050    
051      public double mean() {
052        return runningMean;
053      }
054    
055      public long min() {
056        return minValue;
057      }
058    
059      public long max() {
060        return maxValue;
061      }
062    
063      public long range() {
064        return maxValue - minValue;
065      }
066    
067      public long populationSize() {
068        return populationSize;
069      }
070    
071      @Override
072      public String toString() {
073        return String.format("Mean: %f, Min: %d, Max: %d, Range: %d, Stddev: %f, Variance: %f",
074            mean(), min(), max(), range(), standardDeviation(), variance());
075      }
076    }
