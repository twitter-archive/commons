001    package com.twitter.common.stats;
002    
003    /**
004     * Precision expresses the maximum epsilon tolerated for a typical size of input
005     * e.g.: Precision(0.01, 1000) express that we tolerate a error of 1% for 1000 entries
006     *       it means that max difference between the real quantile and the estimate one is
007     *       error = 0.01*1000 = 10
008     *       For an entry like (1 to 1000), q(0.5) will be [490 <= x <= 510] (real q(0.5) = 500)
009     */
010    public class Precision {
011      private final double epsilon;
012      private final int n;
013    
014      /**
015       * Create a Precision instance representing a precision per number of entries
016       *
017       * @param epsilon is the maximum error tolerated
018       * @param n size of the data set
019       */
020      public Precision(double epsilon, int n) {
021        this.epsilon = epsilon;
022        this.n = n;
023      }
024    
025      public double getEpsilon() {
026        return epsilon;
027      }
028    
029      public int getN() {
030        return n;
031      }
032    }
