018    package com.twitter.common.stats;
019    
020    import java.util.Map;
021    
022    /**
023     * Same as CounterMap<K>, but also keeps track of the item with the highest count.
024     */
025    public class CounterMapWithTopKey<K> extends CounterMap<K> {
026    
027      private K mostCommonKey = null;
028    
029      /**
030       * Updates the most common key, if needed.
031       *
032       * @param key The key to check.
033       * @param count The count for the key.
034       * @return The count.
035       */
036      private int updateMostCommon(K key, int count) {
037        if (count > get(mostCommonKey)) {
038          mostCommonKey = key;
039        }
040        return count;
041      }
042    
043      /**
044       * Increments the counter value associated with {@code key}, and returns the new value.
045       *
046       * @param key The key to increment
047       * @return The incremented value.
048       */
049      @Override
050      public int incrementAndGet(K key) {
051        return updateMostCommon(key, super.incrementAndGet(key));
052      }
053    
054      /**
055       * Assigns a value to a key.
056       *
057       * @param key The key to assign a value to.
058       * @param newValue The value to assign.
059       */
060      @Override
061      public void set(K key, int newValue) {
062        super.set(key, updateMostCommon(key, newValue));
063      }
064    
065      /**
066       * Resets the value for {@code key}.  This will simply set the stored value to 0.
067       * The most common key is updated by scanning the entire map.
068       *
069       * @param key The key to reset.
070       */
071      @Override
072      public void reset(K key) {
073        super.reset(key);
074        for (Map.Entry<K, Integer> entry : this) {
075          updateMostCommon(entry.getKey(), entry.getValue());
076        }
077      }
078    
079      /**
080       *
081       * @return The key with the highest count in the map. If multiple keys have this count, return
082       * an arbitrary one.
083       */
084      public K getMostCommonKey() {
085        return mostCommonKey;
086      }
087    
088      @Override
089      public String toString() {
090        return new StringBuilder(super.toString()).append(String.format("Most common key: %s\n",
091            mostCommonKey.toString())).toString();
092      }
093    }
