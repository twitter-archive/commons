017    package com.twitter.common.base;
018    
019    import java.lang.reflect.Constructor;
020    import java.lang.reflect.InvocationTargetException;
021    import java.lang.reflect.Modifier;
022    
023    import com.google.common.base.Preconditions;
024    
025    /**
026     * Utility methods for working with Suppliers.
027     *
028     * @author John Sirois
029     */
030    public final class MoreSuppliers {
031    
032      private MoreSuppliers() {
033        // utility
034      }
035    
036      /**
037       * Creates a Supplier that uses the no-argument constructor of {@code type} to supply new
038       * instances.
039       *
040       * @param type the type of object this supplier creates
041       * @param <T> the type of object this supplier creates
042       * @return a Supplier that created a new obeject of type T on each call to {@link Supplier#get()}
043       * @throws IllegalArgumentException if the given {@code type} does not have a no-arg constructor
044       */
045      public static <T> Supplier<T> of(final Class<? extends T> type) {
046        Preconditions.checkNotNull(type);
047    
048        try {
049          final Constructor<? extends T> constructor = getNoArgConstructor(type);
050          return new Supplier<T>() {
051            @Override public T get() {
052              try {
053                return constructor.newInstance();
054              } catch (InstantiationException e) {
055                throw instantiationFailed(e, type);
056              } catch (IllegalAccessException e) {
057                throw instantiationFailed(e, type);
058              } catch (InvocationTargetException e) {
059                throw instantiationFailed(e, type);
060              }
061            }
062          };
063        } catch (NoSuchMethodException e) {
064          throw new IllegalArgumentException("No accessible no-arg constructor for " + type, e);
065        }
066      }
067    
068      private static RuntimeException instantiationFailed(Exception cause, Object type) {
069        return new RuntimeException("Could not create a new instance of type: " + type, cause);
070      }
071    
072      private static <T> Constructor<T> getNoArgConstructor(Class<T> type)
073        throws NoSuchMethodException {
074    
075        try {
076          Constructor<T> constructor = type.getConstructor();
077          if (!MoreSuppliers.class.getPackage().equals(type.getPackage())
078              && !Modifier.isPublic(type.getModifiers())) {
079            // Handle a public no-args constructor in a non-public class
080            constructor.setAccessible(true);
081          }
082          return constructor;
083        } catch (NoSuchMethodException e) {
084          Constructor<T> declaredConstructor = type.getDeclaredConstructor();
085          declaredConstructor.setAccessible(true);
086          return declaredConstructor;
087        }
088      }
089    }
