017    package com.twitter.common.base;
018    
019    import com.google.common.base.Function;
020    import com.google.common.base.Predicate;
021    import com.google.common.base.Predicates;
022    import com.google.common.base.Throwables;
023    import com.google.common.collect.ImmutableList;
024    import com.google.common.collect.Iterables;
025    
026    import static com.google.common.base.Preconditions.checkArgument;
027    import static com.google.common.base.Preconditions.checkNotNull;
028    
029    /**
030     * Utilities for dealing with Closures.
031     *
032     * @author John Sirois
033     */
034    public final class Closures {
035    
036      private static final Closure NOOP = new Closure() {
037        @Override public void execute(Object item) {
038          // noop
039        }
040      };
041    
042      private Closures() {
043        // utility
044      }
045    
046      /**
047       * Converts a closure into a function returning {@code null}.
048       */
049      public static <T> Function<T, Void> asFunction(final ExceptionalClosure<T, ?> closure) {
050        checkNotNull(closure);
051    
052        return new Function<T, Void>() {
053          @Override public Void apply(T item) {
054            try {
055              closure.execute(item);
056            } catch (Exception e) {
057              Throwables.propagate(e);
058            }
059            return null;
060          }
061        };
062      }
063    
064      /**
065       * Varargs equivalent of {@link #combine(Iterable)}.
066       *
067       * @param closures Closures to combine.
068       * @param <T> Type accepted by the closures.
069       * @return A single closure that will fan out all calls to {@link Closure#execute(Object)} to
070       *    the wrapped closures.
071       */
072      public static <T> Closure<T> combine(Closure<T>... closures) {
073        return combine(ImmutableList.copyOf(closures));
074      }
075    
076      /**
077       * Combines multiple closures into a single closure, whose calls are replicated sequentially
078       * in the order that they were provided.
079       * If an exception is encountered from a closure it propagates to the top-level closure and the
080       * remaining closures are not executed.
081       *
082       * @param closures Closures to combine.
083       * @param <T> Type accepted by the closures.
084       * @return A single closure that will fan out all calls to {@link Closure#execute(Object)} to
085       *    the wrapped closures.
086       */
087      public static <T> Closure<T> combine(Iterable<Closure<T>> closures) {
088        checkNotNull(closures);
089        checkArgument(Iterables.all(closures, Predicates.notNull()));
090    
091        final Iterable<Closure<T>> closuresCopy = ImmutableList.copyOf(closures);
092    
093        return new Closure<T>() {
094          @Override public void execute(T item) {
095            for (Closure<T> closure : closuresCopy) {
096              closure.execute(item);
097            }
098          }
099        };
100      }
101    
102      /**
103       * Applies a filter to a closure, such that the closure will only be called when the filter is
104       * satisfied (returns {@code true}}.
105       *
106       * @param filter Filter to determine when {@code closure} is called.
107       * @param closure Closure to filter.
108       * @param <T> Type handled by the filter and the closure.
109       * @return A filtered closure.
110       */
111      public static <T> Closure<T> filter(final Predicate<T> filter, final Closure<T> closure) {
112        checkNotNull(filter);
113        checkNotNull(closure);
114    
115        return new Closure<T>() {
116          @Override public void execute(T item) {
117            if (filter.apply(item)) {
118              closure.execute(item);
119            }
120          }
121        };
122      }
123    
124      /**
125       * Returns a closure that will do nothing.
126       *
127       * @param <T> The closure argument type.
128       * @return A closure that does nothing.
129       */
130      @SuppressWarnings("unchecked")
131      public static <T> Closure<T> noop() {
132        return NOOP;
133      }
134    }
