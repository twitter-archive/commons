045    /**
046     * A simple dependency tracking compiler that maps generated classes to the owning sources.
047     *
048     * Supports a <pre>-dependencyfile</pre> option to output dependency information to in the form:
049     * <pre>
050     * [source file path] -&gt; [class file path]
051     * </pre>
052     *
053     * There may be multiple lines per source file if the file contains multiple top level classes or
054     * inner classes.  All paths are normalized to be relative to the classfile output directory.
055     */
056    // SUPPRESS CHECKSTYLE HideUtilityClassConstructor
057    public final class Compiler {
058    
059      public static final String DEPENDENCYFILE_FLAG = "-dependencyfile";
060    
061      /**
062       * A file manager that intercepts requests for class output files to track dependencies.
063       */
064      static final class DependencyTrackingFileManager
065          extends ForwardingJavaFileManager<StandardJavaFileManager> {
066    
067        private final LinkedHashMap<String, List<String>> sourceToClasses =
068            new LinkedHashMap<String, List<String>>();
069        private final Set<String> priorSources = new HashSet<String>();
070        private final File dependencyFile;
071    
072        private List<String> outputPath;
073        private File outputDir;
074    
075        DependencyTrackingFileManager(StandardJavaFileManager fileManager, File dependencies)
076            throws IOException {
077    
078          super(fileManager);
079          this.dependencyFile = dependencies;
080    
081          if (dependencyFile.exists()) {
082            System.out.println("Reading existing dependency file at " + dependencies);
083            BufferedReader dependencyReader = new BufferedReader(new FileReader(dependencies));
084            try {
085              int line = 0;
086              while (true) {
087                String mapping = dependencyReader.readLine();
088                if (mapping == null) {
089                  break;
090                }
091    
092                line++;
093                String[] components = mapping.split(" -> ");
094                if (components.length != 2) {
095                  System.err.printf("Ignoring malformed dependency in %s[%d]: %s\n",
096                      dependencies, line, mapping);
097                } else {
098                  String sourceRelpath = components[0];
099                  String classRelpath = components[1];
100                  addMapping(sourceRelpath, classRelpath);
101                }
102              }
103            } finally {
104              dependencyReader.close();
105            }
106          }
107          priorSources.addAll(sourceToClasses.keySet());
108        }
109    
110        @Override
111        public JavaFileObject getJavaFileForOutput(Location location, String className, Kind kind,
112            FileObject sibling) throws IOException {
113    
114          JavaFileObject file = super.getJavaFileForOutput(location, className, kind, sibling);
115          if (Kind.CLASS == kind) {
116            addMapping(toOutputRelpath(sibling), toOutputRelpath(file));
117          }
118          return file;
119        }
120    
121        private void addMapping(String sourceFile, String classFile) {
122          List<String> classFiles = sourceToClasses.get(sourceFile);
123          if (classFiles == null || priorSources.remove(sourceFile)) {
124            classFiles = new ArrayList<String>();
125            sourceToClasses.put(sourceFile, classFiles);
126          }
127          classFiles.add(classFile);
128        }
129    
130        private String toOutputRelpath(FileObject file) {
131          List<String> base = new ArrayList<String>(getOutputPath());
132          List<String> path = toList(file);
133          for (Iterator<String> baseIter = base.iterator(), pathIter = path.iterator();
134               baseIter.hasNext() && pathIter.hasNext();) {
135            if (!baseIter.next().equals(pathIter.next())) {
136              break;
137            } else {
138              baseIter.remove();
139              pathIter.remove();
140            }
141          }
142    
143          if (!base.isEmpty()) {
144            path.addAll(0, Collections.nCopies(base.size(), ".."));
145          }
146          return join(path);
147        }
148    
149        private String join(List<String> components) {
150          StringBuilder path = new StringBuilder();
151          for (int i = 0, max = components.size(); i < max; i++) {
152            if (i > 0) {
153              path.append(File.separatorChar);
154            }
155            path.append(components.get(i));
156          }
157          return path.toString();
158        }
159    
160        private List<String> toList(FileObject path) {
161          return new ArrayList<String>(Arrays.asList(path.toUri().normalize().getPath().split("/")));
162        }
163    
164        private synchronized List<String> getOutputPath() {
165          if (outputPath == null) {
166            for (File path : fileManager.getLocation(StandardLocation.CLASS_OUTPUT)) {
167              if (outputPath != null) {
168                throw new IllegalStateException("Expected exactly 1 output path");
169              }
170              List<String> components = new ArrayList<String>();
171              File f = path;
172              while (f != null) {
173                components.add(f.getName());
174                f = f.getParentFile();
175              }
176              Collections.reverse(components);
177              outputPath = components;
178              outputDir = path;
179            }
180          }
181          return outputPath;
182        }
183    
184        @Override
185        public void close() throws IOException {
186          super.close();
187    
188          System.out.println("Writing class dependency file to " + dependencyFile);
189          PrintWriter dependencyWriter = new PrintWriter(new FileWriter(dependencyFile, false));
190          for (Entry<String, List<String>> entry : sourceToClasses.entrySet()) {
191            String sourceFile = entry.getKey();
192            for (String classFile : entry.getValue()) {
193              if (!priorSources.contains(sourceFile)
194                  || (new File(outputDir, sourceFile).exists()
195                      && new File(outputDir, classFile).exists())) {
196                dependencyWriter.printf("%s -> %s\n", sourceFile, classFile);
197              }
198            }
199          }
200          dependencyWriter.close();
201        }
202      }
203    
204      /**
205       * Should not be used; instead invoke {@link #main} directly.  Only present to conform to
206       * a common compiler interface idiom expected by jmake.
207       */
208      public Compiler() {
209      }
210    
211      /**
212       * Passes through all args to the system java compiler and tracks classes generated for each
213       * source file.
214       *
215       * @param args The command line arguments.
216       * @return An exit code where 0 indicates successful compilation.
217       * @throws IOException If there is a problem writing the dependency file.
218       */
219      public static int compile(String[] args) throws IOException {
220        JavaCompiler compiler = ToolProvider.getSystemJavaCompiler();
221        StandardJavaFileManager standardFileManager =
222            compiler.getStandardFileManager(
223                null, // use default diagnostic listener
224                null, // default locale
225                null); // default charset
226    
227        List<String> options = new ArrayList<String>();
228        List<String> compilationUnits = new ArrayList<String>();
229        File dependencyFile = null;
230        for (Iterator<String> iter = Arrays.asList(args).iterator(); iter.hasNext();) {
231          String arg = iter.next();
232          if (DEPENDENCYFILE_FLAG.equals(arg)) {
233            if (!iter.hasNext()) {
234              System.err.printf("%s requires an argument specifying the output path\n",
235                  DEPENDENCYFILE_FLAG);
236              return 1;
237            }
238            dependencyFile = new File(iter.next());
239          } else if (arg.startsWith("-")) {
240            int argCount = compiler.isSupportedOption(arg);
241            if (argCount == -1) {
242              argCount = standardFileManager.isSupportedOption(arg);
243            }
244            if (argCount == -1) {
245              System.err.println("WARNING: Skipping unsupported option " + arg);
246            } else {
247              options.add(arg);
248              while (argCount-- > 0) {
249                if (iter.hasNext()) {
250                  options.add(iter.next());
251                }
252              }
253            }
254          } else {
255            compilationUnits.add(arg);
256          }
257        }
258    
259        if (dependencyFile == null) {
260          // Nothing special to do - just pass through and use the default output streams.
261          return compiler.run(null, null, null, args);
262        }
263    
264        DependencyTrackingFileManager fileManager =
265            new DependencyTrackingFileManager(standardFileManager, dependencyFile);
266        try {
267          CompilationTask compilationTask =
268              compiler.getTask(
269                  null, // use default output stream
270                  fileManager,
271                  null, // use default diagnostic listener
272                  options,
273                  null, // we specify no custom annotation processors manually here
274                  standardFileManager.getJavaFileObjectsFromStrings(compilationUnits));
275    
276          boolean success = compilationTask.call();
277          return success ? 0 : 1;
278        } finally {
279          fileManager.close();
280        }
281      }
282    
283      /**
284       * Passes through all args to the system java compiler and tracks classes generated for each
285       * source file.
286       *
287       * @param args The command line arguments.
288       * @throws IOException If there is a problem writing the dependency file.
289       */
290      public static void main(String[] args) throws IOException {
291        exit(compile(args));
292      }
293    
294      private static void exit(int code) {
295        // We're a main - its fine to exit.
296        // SUPPRESS CHECKSTYLE RegexpSinglelineJava
297        System.exit(code);
298      }
