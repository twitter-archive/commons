017    package com.twitter.common.zookeeper.testing;
018    
019    import java.io.IOException;
020    
021    import com.google.common.testing.TearDown;
022    import com.google.common.testing.junit4.TearDownTestCase;
023    
024    import org.junit.Before;
025    
026    import com.twitter.common.application.ActionController;
027    import com.twitter.common.quantity.Amount;
028    import com.twitter.common.quantity.Time;
029    import com.twitter.common.zookeeper.ZooKeeperClient;
030    import com.twitter.common.zookeeper.ZooKeeperClient.ZooKeeperConnectionException;
031    
032    /**
033     * A baseclass for in-process zookeeper tests.
034     * Uses ZooKeeperTestHelper to start the server and create clients: new tests should directly use
035     * that helper class instead of extending this class.
036     *
037     * @author John Sirois
038     */
039    public abstract class BaseZooKeeperTest extends TearDownTestCase {
040      private ZooKeeperTestServer zkTestServer;
041    
042      @Before
043      public final void setUp() throws Exception {
044        final ActionController shutdownRegistry = new ActionController();
045        addTearDown(new TearDown() {
046          @Override public void tearDown() {
047            shutdownRegistry.execute();
048          }
049        });
050        zkTestServer = new ZooKeeperTestServer(0, shutdownRegistry);
051        zkTestServer.startNetwork();
052      }
053    
054      /**
055       * Starts zookeeper back up on the last used port.
056       */
057      protected final void restartNetwork() throws IOException, InterruptedException {
058        zkTestServer.restartNetwork();
059      }
060    
061      /**
062       * Shuts down the in-process zookeeper network server.
063       */
064      protected final void shutdownNetwork() {
065        zkTestServer.shutdownNetwork();
066      }
067    
068      /**
069       * Expires the active session for the given client.  The client should be one returned from
070       * {@link #createZkClient}.
071       *
072       * @param zkClient the client to expire
073       * @throws ZooKeeperClient.ZooKeeperConnectionException if a problem is encountered connecting to
074       *    the local zk server while trying to expire the session
075       * @throws InterruptedException if interrupted while requesting expiration
076       */
077      protected final void expireSession(ZooKeeperClient zkClient)
078          throws ZooKeeperConnectionException, InterruptedException {
079        zkTestServer.expireClientSession(zkClient);
080      }
081    
082      /**
083       * Returns the current port to connect to the in-process zookeeper instance.
084       */
085      protected final int getPort() {
086        return zkTestServer.getPort();
087      }
088    
089      /**
090       * Returns a new zookeeper client connected to the in-process zookeeper server with the
091       * default session timeout in ZooKeeperTestHelper.
092       */
093      protected final ZooKeeperClient createZkClient() {
094        return zkTestServer.createClient();
095      }
096    
097      /**
098       * Returns a new zookeeper client connected to the in-process zookeeper server with a custom
099       * {@code sessionTimeout}.
100       */
101      protected final ZooKeeperClient createZkClient(Amount<Integer, Time> sessionTimeout) {
102        return zkTestServer.createClient(sessionTimeout);
103      }
104    }
