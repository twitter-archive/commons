018    package com.twitter.common.zookeeper.testing;
019    
020    import java.io.IOException;
021    
022    import com.google.common.testing.TearDown;
023    import com.google.common.testing.junit4.TearDownTestCase;
024    
025    import org.junit.Before;
026    
027    import com.twitter.common.application.ActionController;
028    import com.twitter.common.quantity.Amount;
029    import com.twitter.common.quantity.Time;
030    import com.twitter.common.zookeeper.ZooKeeperClient;
031    import com.twitter.common.zookeeper.ZooKeeperClient.ZooKeeperConnectionException;
032    
033    /**
034     * A baseclass for in-process zookeeper tests.
035     * Uses ZooKeeperTestHelper to start the server and create clients: new tests should directly use
036     * that helper class instead of extending this class.
037     *
038     * @author John Sirois
039     */
040    public abstract class BaseZooKeeperTest extends TearDownTestCase {
041      private ZooKeeperTestServer zkTestServer;
042    
043      @Before
044      public final void setUp() throws Exception {
045        final ActionController shutdownRegistry = new ActionController();
046        addTearDown(new TearDown() {
047          @Override public void tearDown() {
048            shutdownRegistry.execute();
049          }
050        });
051        zkTestServer = new ZooKeeperTestServer(0, shutdownRegistry);
052        zkTestServer.startNetwork();
053      }
054    
055      /**
056       * Starts zookeeper back up on the last used port.
057       */
058      protected final void restartNetwork() throws IOException, InterruptedException {
059        zkTestServer.restartNetwork();
060      }
061    
062      /**
063       * Shuts down the in-process zookeeper network server.
064       */
065      protected final void shutdownNetwork() {
066        zkTestServer.shutdownNetwork();
067      }
068    
069      /**
070       * Expires the active session for the given client.  The client should be one returned from
071       * {@link #createZkClient}.
072       *
073       * @param zkClient the client to expire
074       * @throws ZooKeeperClient.ZooKeeperConnectionException if a problem is encountered connecting to
075       *    the local zk server while trying to expire the session
076       * @throws InterruptedException if interrupted while requesting expiration
077       */
078      protected final void expireSession(ZooKeeperClient zkClient)
079          throws ZooKeeperConnectionException, InterruptedException {
080        zkTestServer.expireClientSession(zkClient);
081      }
082    
083      /**
084       * Returns the current port to connect to the in-process zookeeper instance.
085       */
086      protected final int getPort() {
087        return zkTestServer.getPort();
088      }
089    
090      /**
091       * Returns a new zookeeper client connected to the in-process zookeeper server with the
092       * default session timeout in ZooKeeperTestHelper.
093       */
094      protected final ZooKeeperClient createZkClient() {
095        return zkTestServer.createClient();
096      }
097    
098      /**
099       * Returns a new zookeeper client connected to the in-process zookeeper server with a custom
100       * {@code sessionTimeout}.
101       */
102      protected final ZooKeeperClient createZkClient(Amount<Integer, Time> sessionTimeout) {
103        return zkTestServer.createClient(sessionTimeout);
104      }
105    }
