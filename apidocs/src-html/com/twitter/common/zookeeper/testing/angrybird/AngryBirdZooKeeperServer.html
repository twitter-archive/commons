028    /**
029     * ZooKeeper server harness for fault testing.
030     *
031     * You may expire sessions directly by SessionId, by Endpoint (host, port pair) or
032     * by leader/follower in a group of ephemeral/sequential nodes.
033     *
034     * The Endpoint encoding currently assumes that the format of the content in the znode is an
035     * "id@host:port" string.
036     */
037    public class AngryBirdZooKeeperServer extends ZooKeeperTestServer {
038    
039      private static final Logger LOG = Logger.getLogger(AngryBirdZooKeeperServer.class.getName());
040      private static final Splitter AT_SPLITTER = Splitter.on('@').trimResults().omitEmptyStrings();
041    
042      public AngryBirdZooKeeperServer(int port, ShutdownRegistry shutdownRegistry) throws IOException {
043        super(port, shutdownRegistry);
044      }
045    
046      /**
047       * Expires the zookeeper session id.
048       *
049       * @param sessionId The Zookeeper session id
050       * @return the sessionId if the session was successfully closed
051       */
052      public final Optional<Long> expireSession(Long sessionId) {
053        return closeSession(Optional.of(sessionId));
054      }
055    
056      /**
057       * Expires the zookeeper session of the given endpoint.
058       * For now, this only supports those endpoints that store their host:port in the znode.
059       *
060       * @param host host address of the endpoint stored in the znode
061       * @param port port of the endpoint stored in the znode
062       * @return the session id of the znode that matches the endpoint if a match is found
063       */
064      public final Optional<Long> expireEndpoint(String host, int port) {
065        return closeSession(getSessionIdFromHostPair(host, port));
066      }
067    
068      /**
069       * Expires zookeeper session of leader rooted at a znode.
070       *
071       * @param path the zookeeper path that is used for leader election
072       * @return the session id of the matching candidate if a match is found
073       */
074      public final Optional<Long> expireLeader(String path) {
075        return closeSession(getLeaderSessionIdFromPath(path));
076      }
077    
078      /**
079       * Expires zookeeper session of follower rooted at a znode.
080       *
081       * @param path the zookeeper path that is used for leader election
082       * @param nodeId (optional) expire a specific follower node if found
083       * @return the session id of the matching candidate if a match is found
084       */
085      public final Optional<Long> expireFollower(String path, Optional<Integer> nodeId) {
086        return closeSession(getFollowerSessionIdFromPath(path, nodeId));
087      }
088    
089      private Optional<Long> closeSession(Optional<Long> sessionId) {
090        if (!sessionId.isPresent()) {
091          LOG.warning("No session found for expiration!");
092        } else {
093          LOG.info("Closing session: " + sessionId);
094          zooKeeperServer.closeSession(sessionId.get().longValue());
095        }
096    
097        return sessionId;
098      }
099    
100      /**
101       * Returns the session whose corresponding znode encodes "host:port"
102       *
103       * @param host ip address of the endpoint
104       * @param port endpoint port
105       * @return session id of the corresponding zk session if a match is found.
106       */
107      private Optional<Long> getSessionIdFromHostPair(String host, int port) {
108        // TODO(vinod): Instead of (host, port) args use the more generic byte[] as args
109        // so that comparison can be made on znodes that are ServerSet ephemerals
110        ZKDatabase zkDb = zooKeeperServer.getZKDatabase();
111    
112        for (long sessionId : zkDb.getSessions()) {
113          for (String path: zkDb.getEphemerals(sessionId)) {
114            LOG.info("SessionId:" + sessionId + " Path:" + path);
115            try {
116              String data = new String(zkDb.getData(path, new Stat(), null));
117              LOG.info("Data in znode: " + data);
118    
119              TestEndpoint endpoint = parseEndpoint(data);
120              LOG.info("Extracted endpoint " + endpoint);
121    
122              if (endpoint.getHost().equals(host) && endpoint.getPort() == port) {
123                LOG.info(String.format(
124                    "Matching session id %s found for endpoint %s:%s", sessionId, host, port));
125                return Optional.of(sessionId);
126              }
127            } catch (NoNodeException e) {
128              LOG.severe("Exception getting data for Path:" + path + " : " + e);
129            } catch (ParseException e) {
130              LOG.severe("Exception parsing data: " + e);
131            } catch (NumberFormatException e) {
132              LOG.severe("Exception in url format " + e);
133            }
134          }
135        }
136    
137        return Optional.absent();
138      }
139    
140      /**
141       * Return the session id of the leader candidate.
142       * See http://zookeeper.apache.org/doc/trunk/recipes.html#sc_leaderElection
143       *
144       * @param zkPath Znode path prefix of the candidates
145       * @return the session id of the corresponding zk session if a match is found.
146       */
147      private Optional<Long> getLeaderSessionIdFromPath(String zkPath) {
148        ZKDatabase zkDb = zooKeeperServer.getZKDatabase();
149        Long leaderSessionId = null;
150        Long masterSeq = Long.MAX_VALUE;
151    
152        // Reg-ex pattern for sequence numbers in znode paths.
153        Pattern pattern = Pattern.compile("\\d+$");
154    
155        // First find the session id of the leading scheduler.
156        for (long sessionId : zkDb.getSessions()) {
157          for (String path: zkDb.getEphemerals(sessionId)) {
158            if (StringUtils.startsWith(path, zkPath)) {
159              try {
160                // Get the sequence number.
161                Matcher matcher = pattern.matcher(path);
162                if (matcher.find()) {
163                  LOG.info("Pattern matched path: " + path + " session: " + sessionId);
164                  Long seq = Long.parseLong(matcher.group());
165                  if (seq < masterSeq) {
166                    masterSeq = seq;
167                    leaderSessionId = sessionId;
168                  }
169                }
170              } catch (NumberFormatException e) {
171                LOG.severe("Exception formatting sequence number " + e);
172              }
173            }
174          }
175        }
176    
177        if (leaderSessionId != null) {
178          LOG.info(String.format("Found session leader for %s: %s", zkPath, leaderSessionId));
179        }
180    
181        return Optional.of(leaderSessionId);
182      }
183    
184      /**
185       * Return the session id of a follower candidate
186       *
187       * @param zkPath Znode path prefix of the candidates
188       * @param candidateId (optional) specific candidate id of follower to expire, otherwise random.
189       * @return session id of the corresponding zk session if a match is found
190       */
191      private Optional<Long> getFollowerSessionIdFromPath(String zkPath, Optional<Integer> nodeId) {
192        Optional<Long> leaderSessionId = getLeaderSessionIdFromPath(zkPath);
193        if (!leaderSessionId.isPresent()) {
194          return leaderSessionId;
195        }
196    
197        ZKDatabase zkDb = zooKeeperServer.getZKDatabase();
198    
199        for (long sessionId : zkDb.getSessions()) {
200          if (sessionId == leaderSessionId.get()) {
201            continue;
202          }
203          for (String path: zkDb.getEphemerals(sessionId)) {
204            if (StringUtils.startsWith(path, zkPath)) {
205              LOG.info(String.format("Found session follower for %s: %s", zkPath, sessionId));
206              if (!nodeId.isPresent()) {
207                return Optional.of(sessionId);
208              } else {
209                TestEndpoint endpoint;
210                try {
211                  endpoint = parseEndpoint(new String(zkDb.getData(path, new Stat(), null)));
212                  if (endpoint.getNodeId() == nodeId.get()) {
213                    return Optional.of(sessionId);
214                  }
215                } catch (ParseException e) {
216                  LOG.severe("Failed to parse endpoint " + path + ": " + e);
217                } catch (NoNodeException e) {
218                  LOG.severe("Exception getting data for Path:" + path + " :" + e);
219                }
220              }
221            }
222          }
223        }
224    
225        return Optional.absent();
226      }
227    
228      private TestEndpoint parseEndpoint(String data) throws ParseException {
229        ImmutableList<String> endpointComponents = ImmutableList.copyOf(AT_SPLITTER.split(data));
230        if (endpointComponents.size() != 2) {
231          throw new ParseException("Unknown znode data: Expected format id@host:port", 0);
232        }
233    
234        int nodeId = Integer.parseInt(endpointComponents.get(0));
235        HostAndPort pair;
236        try {
237          pair = HostAndPort.fromString(endpointComponents.get(1));
238        } catch (IllegalArgumentException e) {
239          throw new ParseException("Failed to parse endpoint data: " + endpointComponents.get(1),
240              data.indexOf('@'));
241        }
242    
243        TestEndpoint endpoint = new TestEndpoint();
244        endpoint.setNodeId(nodeId);
245        endpoint.setHost(pair.getHostText());
246        endpoint.setPort(pair.getPort());
247        return endpoint;
248      }
