017    package com.twitter.common.application.http;
018    
019    import com.google.inject.Binder;
020    
021    /**
022     * A utility class to register the file resources for the graph viewer.
023     *
024     * @author William Farner
025     */
026    public final class GraphViewer {
027    
028      private GraphViewer() {
029        // Utility class.
030      }
031    
032      /**
033       * Registers required resources with the binder.
034       *
035       * @param binder Binder to register with.
036       */
037      public static void registerResources(Binder binder) {
038        Registration.registerHttpAsset(binder, "/graphview/dygraph-combined.js", GraphViewer.class,
039            "dygraph-combined.js", "application/javascript", true);
040        Registration.registerHttpAsset(binder,
041            "/graphview", GraphViewer.class, "graphview.html", "text/html", false);
042      }
043    }
