001    package com.twitter.common.metrics;
002    
003    import java.util.LinkedList;
004    
005    import com.google.common.annotations.VisibleForTesting;
006    import com.google.common.base.Preconditions;
007    import com.google.common.base.Supplier;
008    import com.google.common.base.Suppliers;
009    import com.google.common.collect.Lists;
010    
011    import com.twitter.common.collections.Pair;
012    import com.twitter.common.quantity.Amount;
013    import com.twitter.common.quantity.Time;
014    import com.twitter.common.util.Clock;
015    
016    /**
017     * Gauge that computes a windowed rate.
018     */
019    public class Rate extends AbstractGauge<Double> {
020    
021      private static final Amount<Long, Time> ONE_SECOND = Amount.of(1L, Time.SECONDS);
022      private static final int MILLIS_PER_SEC = Amount.of(1, Time.SECONDS).as(Time.MILLISECONDS);
023    
024      private final Supplier<? extends Number> valueAccessor;
025      private final LinkedList<Pair<Long, Double>> samples = Lists.newLinkedList();
026      private final Clock clock;
027      private final long windowLengthMs;
028    
029      @VisibleForTesting
030      <T extends Number> Rate(String name, Supplier<T> valueAccessor,
031          Amount<Long, Time> windowLength, Clock clock) {
032        super(name + "_per_" + windowLength.getValue() + windowLength.getUnit().toString());
033        this.valueAccessor = Preconditions.checkNotNull(valueAccessor);
034        this.windowLengthMs = windowLength.as(Time.MILLISECONDS);
035        this.clock = Preconditions.checkNotNull(clock);
036      }
037    
038      /**
039       * Creates a rate using a supplier to access values.
040       *
041       * @param name Name of the rate.
042       * @param valueAccessor Supplier to access values.
043       * @param windowLength Sliding window duration for computing rate.
044       * @param <T> Supplier type.
045       */
046      public <T extends Number> Rate(String name, Supplier<T> valueAccessor,
047          Amount<Long, Time> windowLength) {
048        this(name, valueAccessor, windowLength, Clock.SYSTEM_CLOCK);
049      }
050    
051      /**
052       * Creates a rate of a number.
053       *
054       * @param name Name of the rate.
055       * @param number Rate input.
056       * @return A rate that computes rate(number).
057       */
058      public static Rate of(String name, Number number) {
059        return of(name, number, Clock.SYSTEM_CLOCK);
060      }
061    
062      @VisibleForTesting
063      static Rate of(String name, Number number, Clock clock) {
064        return new Rate(name, Suppliers.ofInstance(number), ONE_SECOND, clock);
065      }
066    
067      /**
068       * Creates a rate of a gauge.
069       *
070       * @param gauge Rate input.
071       * @return A rate that computes rate(gauge).
072       */
073      public static Rate of(Gauge gauge) {
074        return of(gauge, Clock.SYSTEM_CLOCK);
075      }
076    
077      @VisibleForTesting
078      static Rate of(Gauge gauge, Clock clock) {
079        return new Rate(gauge.getName(), Gauges.asSupplier(gauge), ONE_SECOND, clock);
080      }
081    
082      @Override
083      public Double read() {
084        long now = clock.nowMillis();
085        long retainThreshold = now - windowLengthMs;
086        while (!samples.isEmpty() && samples.peekLast().getFirst() < retainThreshold) {
087          samples.removeLast();
088        }
089    
090        double newSample = valueAccessor.get().doubleValue();
091        samples.addFirst(Pair.of(now, newSample));
092    
093        if (samples.size() == 1) {
094          return 0d;
095        }
096    
097        Pair<Long, Double> oldestSample = samples.getLast();
098        double dy = newSample - oldestSample.getSecond();
099        double dt = now - oldestSample.getFirst();
100        return dt == 0 ? 0 : ((dy * MILLIS_PER_SEC) / dt);
101      }
102    }
