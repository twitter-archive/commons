001    package com.twitter.common.metrics;
002    
003    /**
004     * A partial Gauge implementation.
005     *
006     * @param <T> Value type.
007     */
008    public abstract class AbstractGauge<T extends Number> implements Gauge<T> {
009    
010      private final String name;
011    
012      /**
013       * Creates an abstract gauge using the provided name.
014       *
015       * @param name Name of the gauge.
016       */
017      public AbstractGauge(String name) {
018        this.name = name;
019      }
020    
021      @Override
022      public String getName() {
023        return name;
024      }
025    }
