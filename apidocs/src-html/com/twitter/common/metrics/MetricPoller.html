001    package com.twitter.common.metrics;
002    
003    import java.util.concurrent.ScheduledExecutorService;
004    import java.util.concurrent.TimeUnit;
005    import java.util.concurrent.atomic.AtomicBoolean;
006    import java.util.logging.Logger;
007    
008    import com.google.common.base.Preconditions;
009    
010    import com.twitter.common.quantity.Amount;
011    import com.twitter.common.quantity.Time;
012    import com.twitter.common.util.concurrent.ExecutorServiceShutdown;
013    
014    /**
015     * A fixed-rate poller that triggers a {@link MetricSampler}.
016     */
017    public class MetricPoller {
018    
019      private static final Logger LOG = Logger.getLogger(MetricPoller.class.getName());
020    
021      private static final Amount<Long, Time> SHUTDOWN_GRACE_PERIOD = Amount.of(2L, Time.SECONDS);
022    
023      private final MetricSampler sampler;
024      private final ScheduledExecutorService executor;
025    
026      private final AtomicBoolean started = new AtomicBoolean(false);
027      private final AtomicBoolean stopped = new AtomicBoolean(false);
028    
029      /**
030       * Creates a new metric poller.
031       *
032       * @param sampler Sampler to fetch metric values from.
033       * @param executor Executor service to run the sampler with.
034       */
035      public MetricPoller(MetricSampler sampler, ScheduledExecutorService executor) {
036        this.sampler = Preconditions.checkNotNull(sampler);
037        this.executor = Preconditions.checkNotNull(executor);
038      }
039    
040      /**
041       * Initiates the poller.
042       * The poller may only be started once.  If an attempt is made to start a poller when it is
043       * already started, {@link IllegalStateException} will be thrown.
044       *
045       * @param pollInterval Fixed poll rate.
046       */
047      public void start(Amount<Long, Time> pollInterval) {
048        Preconditions.checkNotNull(pollInterval);
049        Preconditions.checkArgument(pollInterval.getValue() > 0, "Poll interval must be positive");
050        Preconditions.checkState(started.compareAndSet(false, true), "Poller is already started.");
051    
052        long pollIntervalMs = pollInterval.as(Time.MILLISECONDS);
053        executor.scheduleAtFixedRate(sampler, pollIntervalMs, pollIntervalMs, TimeUnit.MILLISECONDS);
054      }
055    
056      /**
057       * Stops the poller.
058       * If the poller is already stopped, or is in the process of being stopped, subsequent calls will
059       * throw {@link IllegalStateException}.
060       */
061      public void stop() {
062        if (stopped.compareAndSet(false, true)) {
063          new ExecutorServiceShutdown(executor, SHUTDOWN_GRACE_PERIOD).execute();
064        } else {
065          LOG.warning("Poller is already stopped, subsequent calls ignored.");
066        }
067      }
068    }
