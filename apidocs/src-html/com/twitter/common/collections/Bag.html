017    package com.twitter.common.collections;
018    
019    import java.util.Iterator;
020    import java.util.Map;
021    import java.util.TreeMap;
022    import java.util.TreeSet;
023    
024    /**
025     * Abstract class that extends a Set<E> with the added
026     * functionality it counts how many times an element has
027     * been added to the Set<E> and can filter elements that
028     * don't meet some frequency.
029     *
030     * @deprecated Please use com.google.common.collect.Multiset instead.
031     *
032     * @author Abdur Chowdhury
033     *
034     * @param <E>
035     */
036    @Deprecated
037    public class Bag<E> extends TreeSet<E> {
038      private static final long serialVersionUID = 1L;
039      private Map<E, long[]> countedSet = null;
040      private long totalFrequency;
041    
042      /**
043       * Creates a new empty bag.
044       */
045      public Bag() {
046        countedSet = new TreeMap<E, long[]>();
047        totalFrequency = 0;
048      }
049    
050      @Override
051      public boolean add(E o) {
052        long[] count = countedSet.get(o);
053        if (count == null) {
054          count = new long[1];
055          count[0] = 1;
056          countedSet.put(o, count);
057        } else {
058          count[0]++;
059        }
060        totalFrequency++;
061        return super.add(o);
062      }
063    
064      /**
065       * Returns the frequency an element has been added to the Set<E>
066       * @param o
067       * @return
068       */
069      public long getCount(E o) {
070        long[] count = countedSet.get(o);
071        if (count == null) {
072          return 0;
073        } else {
074          return count[0];
075        }
076      }
077    
078      /**
079       * Returns the total objects added to the Set<E>
080       *
081       * @return
082       */
083      public long getTotalCount() {
084        return totalFrequency;
085      }
086    
087      @Override
088      public Iterator<E> iterator() {
089        return countedSet.keySet().iterator();
090      }
091    
092      @Override
093      public void clear() {
094        super.clear();
095        countedSet.clear();
096        totalFrequency = 0;
097      }
098    
099      /**
100       * Return the number of elements stored in the Bag<E>.
101       */
102      @Override
103      public int size() {
104        return countedSet.size();
105      }
106    
107      /**
108       * Filters objects from the Set<E> that have been added
109       * less than X times.
110       *
111       * @param freq
112       * @return
113       */
114      public int removeIfCountLessThan(int freq) {
115        E key = null;
116        int removed = 0;
117        long count;
118        Iterator<E> it = countedSet.keySet().iterator();
119        while (it.hasNext()) {
120          key = it.next();
121          count = countedSet.get(key)[0];
122          if (count < freq) {
123            it.remove();
124            super.remove(key);
125            removed++;
126            totalFrequency -= count;
127          }
128        }
129        return removed;
130      }
131    
132      /**
133       * Filters objects from the Set<E> that have been added
134       * greater than X times.
135       *
136       * @param freq
137       * @return
138       */
139      public int removeIfCountGreaterThan(int freq) {
140        E key = null;
141        int removed = 0;
142        long count;
143        Iterator<E> it = countedSet.keySet().iterator();
144        while (it.hasNext()) {
145          key = it.next();
146          count = countedSet.get(key)[0];
147          if (count > freq) {
148            it.remove();
149            super.remove(key);
150            removed++;
151            totalFrequency -= count;
152          }
153        }
154        return removed;
155      }
156    
157      @Override
158      public String toString() {
159        StringBuffer sb = new StringBuffer();
160        E key = null;
161        long count;
162        Iterator<E> it = countedSet.keySet().iterator();
163        while (it.hasNext()) {
164          key = it.next();
165          count = countedSet.get(key)[0];
166          sb.append("K:" + key + "\tV:" + count + "\n");
167        }
168        return sb.toString();
169      }
170    
171    }
