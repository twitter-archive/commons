018    package com.twitter.common.collections;
019    
020    import javax.annotation.Nullable;
021    
022    import com.google.common.base.Function;
023    
024    import org.apache.commons.lang.builder.EqualsBuilder;
025    import org.apache.commons.lang.builder.HashCodeBuilder;
026    
027    
028    /**
029     * An immutable 2-tuple with value-equals semantics.
030     *
031     * @param <A> The type of the 1st item in the pair.
032     * @param <B> The type of the 2nd item in the pair.
033     *
034     * @author William Farner
035     */
036    public class Pair<A, B> {
037    
038      @Nullable
039      private final A first;
040      @Nullable
041      private final B second;
042    
043      /**
044       * Creates a new pair.
045       *
046       * @param first The first value.
047       * @param second The second value.
048       */
049      public Pair(@Nullable A first, @Nullable B second) {
050        this.first = first;
051        this.second = second;
052      }
053    
054      @Nullable
055      public A getFirst() {
056        return first;
057      }
058    
059      @Nullable
060      public B getSecond() {
061        return second;
062      }
063    
064      @Override
065      public boolean equals(Object o) {
066        if (o == this) { return true; }
067        if (!(o instanceof Pair)) { return false; }
068    
069        Pair that = (Pair) o;
070        return new EqualsBuilder()
071            .append(this.first, that.first)
072            .append(this.second, that.second)
073            .isEquals();
074      }
075    
076      @Override
077      public String toString() {
078        return String.format("(%s, %s)", getFirst(), getSecond());
079      }
080    
081      @Override
082      public int hashCode() {
083        return new HashCodeBuilder()
084            .append(first)
085            .append(second)
086            .toHashCode();
087      }
088    
089      /**
090       * Creates a function that can extract the first item of pairs of the given type parametrization.
091       *
092       * @param <S> The type of the 1st item in the pair.
093       * @param <T> The type of the 2nd item in the pair.
094       * @return A function that will extract the 1st item in a pair.
095       */
096      public static <S, T> Function<Pair<S, T>, S> first() {
097        return new Function<Pair<S, T>, S>() {
098          @Override public S apply(Pair<S, T> pair) {
099            return pair.first;
100          }
101        };
102      }
103    
104      /**
105       * Creates a function that can extract the second item of pairs of the given type parametrization.
106       *
107       * @param <S> The type of the 1st item in the pair.
108       * @param <T> The type of the 2nd item in the pair.
109       * @return A function that will extract the 2nd item in a pair.
110       */
111      public static <S, T> Function<Pair<S, T>, T> second() {
112        return new Function<Pair<S, T>, T>() {
113          @Override public T apply(Pair<S, T> pair) {
114            return pair.second;
115          }
116        };
117      }
118    
119      /**
120       * Convenience method to create a pair.
121       *
122       * @param a The first value.
123       * @param b The second value.
124       * @param <A> The type of the 1st item in the pair.
125       * @param <B> The type of the 2nd item in the pair.
126       * @return A new pair of [a, b].
127       */
128      public static <A, B> Pair<A, B> of(@Nullable A a, @Nullable B b) {
129        return new Pair<A, B>(a, b);
130      }
131    }
