017    package com.twitter.common.collections;
018    
019    import com.google.common.base.Preconditions;
020    
021    /**
022     * Convenience class for doing bit-level operations on ints and longs.
023     *
024     * @author William Farner
025     */
026    public final class Bits {
027    
028      private static final int LSB = 0;
029      private static final int INT_MSB = 31;
030      private static final int LONG_MSB = 63;
031    
032      private Bits() {
033        // Utility.
034      }
035    
036      /**
037       * Tests whether a bit is set in an int value.
038       *
039       * @param value The bit field to test.
040       * @param bit The index of the bit to test, where bit 0 is the LSB.
041       * @return {@code true} if the bit is set, {@code false} otherwise.
042       */
043      public static boolean isBitSet(int value, int bit) {
044        Preconditions.checkState(bit >= LSB);
045        Preconditions.checkState(bit <= INT_MSB);
046        int mask = 1 << bit;
047        return (value & mask) != 0;
048      }
049    
050      /**
051       * Tests whether a bit is set in a long value.
052       *
053       * @param value The bit field to test.
054       * @param bit The index of the bit to test, where bit 0 is the LSB.
055       * @return {@code true} if the bit is set, {@code false} otherwise.
056       */
057      public static boolean isBitSet(long value, int bit) {
058        Preconditions.checkState(bit >= LSB);
059        Preconditions.checkState(bit <= LONG_MSB);
060        long mask = 1L << bit;
061        return (value & mask) != 0;
062      }
063    
064      /**
065       * Sets a bit in an int value.
066       *
067       * @param value The bit field to modify.
068       * @param bit The index of the bit to set, where bit 0 is the LSB.
069       * @return The original value, with the indexed bit set.
070       */
071      public static int setBit(int value, int bit) {
072        Preconditions.checkState(bit >= LSB);
073        Preconditions.checkState(bit <= INT_MSB);
074        int mask = 1 << bit;
075        return value | mask;
076      }
077    
078      /**
079       * Sets a bit in a long value.
080       *
081       * @param value The bit field to modify.
082       * @param bit The index of the bit to set, where bit 0 is the LSB.
083       * @return The original value, with the indexed bit set.
084       */
085      public static long setBit(long value, int bit) {
086        Preconditions.checkState(bit >= LSB);
087        Preconditions.checkState(bit <= LONG_MSB);
088        long mask = 1L << bit;
089        return value | mask;
090      }
091    
092      /**
093       * Clears a bit in an int value.
094       *
095       * @param value The bit field to modify.
096       * @param bit The index of the bit to clear, where bit 0 is the LSB.
097       * @return The original value, with the indexed bit clear.
098       */
099      public static int clearBit(int value, int bit) {
100        Preconditions.checkState(bit >= LSB);
101        Preconditions.checkState(bit <= INT_MSB);
102        int mask = ~setBit(0, bit);
103        return value & mask;
104      }
105    
106      /**
107       * Clears a bit in a long value.
108       *
109       * @param value The bit field to modify.
110       * @param bit The index of the bit to clear, where bit 0 is the LSB.
111       * @return The original value, with the indexed bit clear.
112       */
113      public static long clearBit(long value, int bit) {
114        Preconditions.checkState(bit >= LSB);
115        Preconditions.checkState(bit <= LONG_MSB);
116        long mask = ~setBit(0L, bit);
117        return value & mask;
118      }
119    }
