001    package com.twitter.common.checkstyle;
002    
003    import java.util.regex.Matcher;
004    import java.util.regex.Pattern;
005    
006    import com.puppycrawl.tools.checkstyle.api.DetailAST;
007    import com.puppycrawl.tools.checkstyle.api.FileContents;
008    import com.puppycrawl.tools.checkstyle.api.TokenTypes;
009    import com.puppycrawl.tools.checkstyle.api.Utils;
010    import com.puppycrawl.tools.checkstyle.checks.javadoc.JavadocMethodCheck;
011    
012    /**
013     * A small extension to the JavadocMethodCheck. Only two additions
014     * are the ability to skip certain methods if they match a regex,
015     * and skip methods if they are short enough (number of non-blank lines)
016     * TODO(Alex Roetter): write a unittest
017     *
018     * @author Alex Roetter
019     */
020    public class JavadocMethodRegexCheck extends JavadocMethodCheck {
021      // Method names that match this pattern do not require javadoc blocks
022      private Pattern methodNameIgnoreRegex;
023      // Don't require javadoc for methods shorter than this many (non-blank)
024      // lines
025      private int minLineCount = -1;
026    
027      // Javadoc not required here due to method length
028      public void setIgnoreMethodNamesRegex(String s) {
029        methodNameIgnoreRegex = Utils.createPattern(s);
030      }
031    
032      public void setMinLineCount(final int n) {
033        minLineCount = n;
034      }
035    
036      /**
037       * Return true if the given method name matches the regex. In that case
038       * we skip enforcement of javadoc for this method
039       */
040      private boolean matchesSkipRegex(final DetailAST aAST) {
041        if (methodNameIgnoreRegex != null) {
042          final DetailAST ident = aAST.findFirstToken(TokenTypes.IDENT);
043          String methodName = ident.getText();
044    
045          Matcher matcher = methodNameIgnoreRegex.matcher(methodName);
046          if (matcher.matches()) {
047            return true;
048          }
049        }
050        return false;
051      }
052    
053      /**
054       * Return true iff the method is short enough that we don't require
055       * a javadoc.
056       */
057      private boolean isShortEnoughToSkip(final DetailAST aAST) {
058        // Based on code from
059        // com/puppycrawl/tools/checkstyle/api/MethodLengthCheck.java
060        final DetailAST openingBrace = aAST.findFirstToken(TokenTypes.SLIST);
061        if (openingBrace != null) {
062          final DetailAST closingBrace =
063            openingBrace.findFirstToken(TokenTypes.RCURLY);
064          int length = closingBrace.getLineNo() - openingBrace.getLineNo() + 1;
065    
066          // skip blank lines
067          final FileContents contents = getFileContents();
068          final int lastLine = closingBrace.getLineNo();
069          for (int i = openingBrace.getLineNo() - 1; i < lastLine; i++) {
070            if (contents.lineIsBlank(i)) {
071              length--;
072            }
073          }
074          if (length <= minLineCount) {
075            return true;
076          }
077        }
078        return false;
079      }
080    
081      @Override
082      protected boolean isMissingJavadocAllowed(final DetailAST aAST) {
083        return super.isMissingJavadocAllowed(aAST)
084          || matchesSkipRegex(aAST)
085          || isShortEnoughToSkip(aAST);
086      }
087    
088      /**
089       * For human readable error messages
090       */
091      @Override
092      protected String getMessageBundle() {
093        return CheckStyleUtils.getMessageBundle(JavadocMethodCheck.class);
094      }
095    }
