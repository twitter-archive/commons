016    package com.twitter.common.checkstyle;
017    
018    import java.util.regex.Matcher;
019    import java.util.regex.Pattern;
020    
021    import com.puppycrawl.tools.checkstyle.api.DetailAST;
022    import com.puppycrawl.tools.checkstyle.api.FileContents;
023    import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024    import com.puppycrawl.tools.checkstyle.api.Utils;
025    import com.puppycrawl.tools.checkstyle.checks.javadoc.JavadocMethodCheck;
026    
027    /**
028     * A small extension to the JavadocMethodCheck. Only two additions
029     * are the ability to skip certain methods if they match a regex,
030     * and skip methods if they are short enough (number of non-blank lines)
031     * TODO(Alex Roetter): write a unittest
032     *
033     * @author Alex Roetter
034     */
035    public class JavadocMethodRegexCheck extends JavadocMethodCheck {
036      // Method names that match this pattern do not require javadoc blocks
037      private Pattern methodNameIgnoreRegex;
038      // Don't require javadoc for methods shorter than this many (non-blank)
039      // lines
040      private int minLineCount = -1;
041    
042      // Javadoc not required here due to method length
043      public void setIgnoreMethodNamesRegex(String s) {
044        methodNameIgnoreRegex = Utils.createPattern(s);
045      }
046    
047      public void setMinLineCount(final int n) {
048        minLineCount = n;
049      }
050    
051      /**
052       * Return true if the given method name matches the regex. In that case
053       * we skip enforcement of javadoc for this method
054       */
055      private boolean matchesSkipRegex(final DetailAST aAST) {
056        if (methodNameIgnoreRegex != null) {
057          final DetailAST ident = aAST.findFirstToken(TokenTypes.IDENT);
058          String methodName = ident.getText();
059    
060          Matcher matcher = methodNameIgnoreRegex.matcher(methodName);
061          if (matcher.matches()) {
062            return true;
063          }
064        }
065        return false;
066      }
067    
068      /**
069       * Return true iff the method is short enough that we don't require
070       * a javadoc.
071       */
072      private boolean isShortEnoughToSkip(final DetailAST aAST) {
073        // Based on code from
074        // com/puppycrawl/tools/checkstyle/api/MethodLengthCheck.java
075        final DetailAST openingBrace = aAST.findFirstToken(TokenTypes.SLIST);
076        if (openingBrace != null) {
077          final DetailAST closingBrace =
078            openingBrace.findFirstToken(TokenTypes.RCURLY);
079          int length = closingBrace.getLineNo() - openingBrace.getLineNo() + 1;
080    
081          // skip blank lines
082          final FileContents contents = getFileContents();
083          final int lastLine = closingBrace.getLineNo();
084          for (int i = openingBrace.getLineNo() - 1; i < lastLine; i++) {
085            if (contents.lineIsBlank(i)) {
086              length--;
087            }
088          }
089          if (length <= minLineCount) {
090            return true;
091          }
092        }
093        return false;
094      }
095    
096      @Override
097      protected boolean isMissingJavadocAllowed(final DetailAST aAST) {
098        return super.isMissingJavadocAllowed(aAST)
099          || matchesSkipRegex(aAST)
100          || isShortEnoughToSkip(aAST);
101      }
102    
103      /**
104       * For human readable error messages
105       */
106      @Override
107      protected String getMessageBundle() {
108        return CheckStyleUtils.getMessageBundle(JavadocMethodCheck.class);
109      }
110    }
