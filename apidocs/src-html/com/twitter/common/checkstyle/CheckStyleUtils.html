017    package com.twitter.common.checkstyle;
018    
019    import com.puppycrawl.tools.checkstyle.api.Check;
020    import com.puppycrawl.tools.checkstyle.api.DetailAST;
021    import com.puppycrawl.tools.checkstyle.api.TokenTypes;
022    
023    /**
024     * Utilities needed by custom checkstyle rules.
025     *
026     * @author Utkarsh Srivastava
027     */
028    public final class CheckStyleUtils {
029    
030      private CheckStyleUtils() {
031        // utility
032      }
033    
034      /**
035       * Returns the message bundle to use for a custom rule. All built-in rules in checkstyle have
036       * a message bundle in the same package as that rule. This method returns the path to that
037       * bundle.
038       *
039       * @param parentBuiltInCheckClass The class of the built-in checkstyle rule that the custom rule
040       *     ultimately derives from.
041       * @return Path to message bundle.
042       */
043      public static String getMessageBundle(final Class<? extends Check> parentBuiltInCheckClass) {
044        String parentBuiltInCheckClassName = parentBuiltInCheckClass.getName();
045        // This code is copied from com.puppycrawl.tools.checkstyle.api.
046        // AbstractViolationReporter.java::getMessageBundle(String s)
047        // If that method were protected (and not package private) we could
048        // just invoke it
049        final int endIndex = parentBuiltInCheckClassName.lastIndexOf('.');
050        final String messages = "messages";
051        if (endIndex < 0) {
052          return messages;
053        }
054        final String packageName = parentBuiltInCheckClassName.substring(0, endIndex);
055        return packageName + "." + messages;
056      }
057    
058      /**
059       * Checks to see if a method has the "@Override" annotation.
060       *
061       * @param aAST The AST to check.
062       * @return Whether the AST represents a method that has the annotation.
063       */
064      public static boolean isOverrideMethod(DetailAST aAST) {
065        // Need it to be a method, cannot have an override on anything else.
066        // Must also have MODIFIERS token to hold the @Override
067        if ((TokenTypes.METHOD_DEF != aAST.getType())
068            || (TokenTypes.MODIFIERS != aAST.getFirstChild().getType())) {
069          return false;
070        }
071    
072        // Now loop over all nodes while they are annotations looking for
073        // an "@Override".
074        DetailAST node = aAST.getFirstChild().getFirstChild();
075        while ((null != node) && (TokenTypes.ANNOTATION == node.getType())) {
076          if ((node.getFirstChild().getType() == TokenTypes.AT)
077              && (node.getFirstChild().getNextSibling().getType() == TokenTypes.IDENT)
078              && ("Override".equals(node.getFirstChild().getNextSibling().getText()))) {
079              return true;
080          }
081          node = node.getNextSibling();
082        }
083        return false;
084      }
085    }
