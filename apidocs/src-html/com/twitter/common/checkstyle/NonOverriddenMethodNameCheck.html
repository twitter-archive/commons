016    package com.twitter.common.checkstyle;
017    
018    import com.puppycrawl.tools.checkstyle.api.AnnotationUtility;
019    import com.puppycrawl.tools.checkstyle.api.DetailAST;
020    import com.puppycrawl.tools.checkstyle.checks.naming.MethodNameCheck;
021    
022    /**
023     * An extension to method name check that does not check-style the name of an overridden method
024     * because the programmer does not have a choice in renaming such methods.
025     * If the interface/base class definition is checkstyled, the bad name will always be caught
026     * over there.
027     *
028     * @author Utkarsh Srivastava
029     */
030    public class NonOverriddenMethodNameCheck extends MethodNameCheck {
031      /**
032       * {@link Override Override} annotation name.
033       */
034      private static final String OVERRIDE = "Override";
035    
036      /**
037       * Fully-qualified {@link Override Override} annotation name.
038       */
039      private static final String FQ_OVERRIDE = "java.lang." + OVERRIDE;
040    
041      // Based on code in com.puppycrawl.tools.checkstyle.checks.annotation.MissingOverrideCheck
042      @Override
043      public void visitToken(DetailAST aAST) {
044        if (!AnnotationUtility.containsAnnotation(aAST, OVERRIDE)
045            && !AnnotationUtility.containsAnnotation(aAST, FQ_OVERRIDE)) {
046          super.visitToken(aAST);
047        }
048      }
049    
050      /**
051       * For human readable error messages
052       */
053      @Override
054      protected String getMessageBundle() {
055        return CheckStyleUtils.getMessageBundle(MethodNameCheck.class);
056      }
057    }
