017    package com.twitter.common.checkstyle;
018    
019    import com.puppycrawl.tools.checkstyle.api.DetailAST;
020    import com.puppycrawl.tools.checkstyle.api.FullIdent;
021    import com.puppycrawl.tools.checkstyle.api.TokenTypes;
022    import com.puppycrawl.tools.checkstyle.checks.coding.AbstractIllegalCheck;
023    
024    /**
025     * Relaxes the built-in checkstyle
026     * {@link com.puppycrawl.tools.checkstyle.checks.coding.IllegalThrowsCheck} by only checking
027     * non-overidden methods.
028     *
029     * @author John Sirois
030     */
031    public class IllegalThrowsCheck extends AbstractIllegalCheck {
032    
033      /**
034       * Creates a new check for illegal throws clauses..
035       */
036      public IllegalThrowsCheck() {
037        super(new String[] {
038          "Error",
039          "RuntimeException",
040          "Throwable",
041          "java.lang.Error",
042          "java.lang.RuntimeException",
043          "java.lang.Throwable",
044        });
045      }
046    
047      @Override
048      public int[] getDefaultTokens() {
049        return new int[] {
050            TokenTypes.METHOD_DEF
051        };
052      }
053    
054      @Override
055      public int[] getRequiredTokens() {
056        return getDefaultTokens();
057      }
058    
059      @Override
060      public void visitToken(DetailAST aDetailAST) {
061        if (!CheckStyleUtils.isOverrideMethod(aDetailAST)) {
062          DetailAST token = aDetailAST.findFirstToken(TokenTypes.LITERAL_THROWS);
063          if (token != null) {
064            checkThrows(token);
065          }
066        }
067      }
068    
069      private void checkThrows(DetailAST throwsProduction) {
070        DetailAST token = throwsProduction.getFirstChild();
071        while (token != null) {
072          if (token.getType() != TokenTypes.COMMA) {
073            final FullIdent ident = FullIdent.createFullIdent(token);
074            if (isIllegalClassName(ident.getText())) {
075              log(token, "illegal.throw", ident.getText());
076            }
077          }
078    
079          token = token.getNextSibling();
080        }
081      }
082    
083      @Override
084      protected String getMessageBundle() {
085        return CheckStyleUtils.getMessageBundle(
086            com.puppycrawl.tools.checkstyle.checks.coding.IllegalThrowsCheck.class);
087      }
088    }
