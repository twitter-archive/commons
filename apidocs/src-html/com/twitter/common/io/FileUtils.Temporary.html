017    package com.twitter.common.io;
018    
019    import java.io.File;
020    import java.io.IOException;
021    import java.util.UUID;
022    
023    import com.google.common.base.Preconditions;
024    
025    import org.apache.commons.lang.SystemUtils;
026    
027    import com.twitter.common.base.ExceptionalClosure;
028    import com.twitter.common.base.ExceptionalFunction;
029    
030    /**
031     * Utility methods for working with files and directories.
032     *
033     * @author John Sirois
034     */
035    public final class FileUtils {
036    
037      /**
038       * A utility for creating and working with temporary files and directories.
039       */
040      public static class Temporary {
041        private static final int MAX_TMP_DIR_TRIES = 5;
042    
043        private final File basedir;
044    
045        /**
046         * Creates a new temporary utility that creates files and directories rooted at {@code basedir}.
047         *
048         * @param basedir The base directory to generate temporary files and directories in.
049         */
050        public Temporary(File basedir) {
051          Preconditions.checkNotNull(basedir);
052          Preconditions.checkArgument(basedir.exists());
053          this.basedir = basedir;
054        }
055    
056        /**
057         * Returns a new empty temporary directory.
058         *
059         * @return a file representing the newly created directory.
060         * @throws IllegalStateException if a new temporary directory could not be created
061         */
062        public File createDir() {
063          File tempDir;
064          int tries = 0;
065          do {
066            // For sanity sake, die eventually if we keep failing to pick a new unique directory name.
067            if (++tries > MAX_TMP_DIR_TRIES) {
068              throw new IllegalStateException("Failed to create a new temp directory in "
069                                              + MAX_TMP_DIR_TRIES + " attempts, giving up");
070            }
071            tempDir = new File(basedir, UUID.randomUUID().toString());
072          } while (!tempDir.mkdir());
073          return tempDir;
074        }
075    
076        /**
077         * Creates a new empty temporary file.
078         *
079         * @return a new empty temporary file
080         * @throws IOException if there was a problem creating a new temporary file
081         */
082        public File createFile() throws IOException {
083          return createFile(".tempfile");
084        }
085    
086        /**
087         * Creates a new empty temporary file with the given filename {@code suffix}.
088         *
089         * @param suffix The suffix for the temporary file name
090         * @return a new empty temporary file
091         * @throws IOException if there was a problem creating a new temporary file
092         */
093        public File createFile(String suffix) throws IOException {
094          return File.createTempFile(FileUtils.class.getName(), suffix, basedir);
095        }
096    
097        /**
098         * Creates a new temporary directory and executes the unit of {@code work} against it ensuring
099         * the directory and its contents are removed after the work completes normally or abnormally.
100         *
101         * @param work The unit of work to execute against the new temporary directory.
102         * @param <E> The type of exception this unit of work can throw.
103         * @throws E bubbled transparently when the unit of work throws
104         */
105        public <E extends Exception> void doWithDir(final ExceptionalClosure<File, E> work)
106            throws E {
107          Preconditions.checkNotNull(work);
108          doWithDir(new ExceptionalFunction<File, Void, E>() {
109            @Override public Void apply(File dir) throws E {
110              work.execute(dir);
111              return null;
112            }
113          });
114        }
115    
116        /**
117         * Creates a new temporary directory and executes the unit of {@code work} against it ensuring
118         * the directory and its contents are removed after the work completes normally or abnormally.
119         *
120         * @param work The unit of work to execute against the new temporary directory.
121         * @param <T> The type of result this unit of work produces.
122         * @param <E> The type of exception this unit of work can throw.
123         * @return the result when the unit of work completes successfully
124         * @throws E bubbled transparently when the unit of work throws
125         */
126        public <T, E extends Exception> T doWithDir(ExceptionalFunction<File, T, E> work)
127            throws E {
128          Preconditions.checkNotNull(work);
129          return doWithTemp(createDir(), work);
130        }
131    
132        /**
133         * Creates a new temporary file and executes the unit of {@code work} against it ensuring
134         * the file is removed after the work completes normally or abnormally.
135         *
136         * @param work The unit of work to execute against the new temporary file.
137         * @param <E> The type of exception this unit of work can throw.
138         * @throws E bubbled transparently when the unit of work throws
139         * @throws IOException if there was a problem creating a new temporary file
140         */
141        public <E extends Exception> void doWithFile(final ExceptionalClosure<File, E> work)
142            throws E, IOException {
143          Preconditions.checkNotNull(work);
144          doWithFile(new ExceptionalFunction<File, Void, E>() {
145            @Override public Void apply(File dir) throws E {
146              work.execute(dir);
147              return null;
148            }
149          });
150        }
151    
152        /**
153         * Creates a new temporary file and executes the unit of {@code work} against it ensuring
154         * the file is removed after the work completes normally or abnormally.
155         *
156         * @param work The unit of work to execute against the new temporary file.
157         * @param <T> The type of result this unit of work produces.
158         * @param <E> The type of exception this unit of work can throw.
159         * @return the result when the unit of work completes successfully
160         * @throws E bubbled transparently when the unit of work throws
161         * @throws IOException if there was a problem creating a new temporary file
162         */
163        public <T, E extends Exception> T doWithFile(ExceptionalFunction<File, T, E> work)
164            throws E, IOException {
165          Preconditions.checkNotNull(work);
166          return doWithTemp(createFile(), work);
167        }
168    
169        private static <T, E extends Exception> T doWithTemp(File file,
170            ExceptionalFunction<File, T, E> work) throws E {
171          try {
172            return work.apply(file);
173          } finally {
174            org.apache.commons.io.FileUtils.deleteQuietly(file);
175          }
176        }
177      }
178    
179      /**
180       * A temporary based at the default system temporary directory.
181       */
182      public static final Temporary SYSTEM_TMP = new Temporary(SystemUtils.getJavaIoTmpDir());
183    
184      /**
185       * Returns a new empty temporary directory.
186       *
187       * @return a file representing the newly created directory.
188       * @throws IllegalStateException if a new temporary directory could not be created
189       */
190      public static File createTempDir() {
191        return SYSTEM_TMP.createDir();
192      }
193    
194      private FileUtils() {
195        // utility
196      }
197    }
