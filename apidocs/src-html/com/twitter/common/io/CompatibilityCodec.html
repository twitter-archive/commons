016    package com.twitter.common.io;
017    
018    import java.io.IOException;
019    import java.io.InputStream;
020    import java.io.OutputStream;
021    import java.io.PushbackInputStream;
022    
023    import com.google.common.base.Preconditions;
024    import com.google.common.base.Predicate;
025    
026    /**
027     * A codec that composes two codecs: a primary and a compatibility codec. It always serializes with
028     * the primary codec, but can make a decision on deserialization based on the first few bytes of the
029     * serialized format whether to use the compatibility codec. This allows for easier transition
030     * between storage formats as the codec remains able to read the old serialized format.
031     *
032     * @author Attila Szegedi
033     *
034     * @param <T> the type of objects this codec is for.
035     */
036    public class CompatibilityCodec<T> implements Codec<T> {
037      private final Codec<T> primaryCodec;
038      private final Codec<T> secondaryCodec;
039      private final int prefixLength;
040      private final Predicate<byte[]> discriminator;
041    
042      private CompatibilityCodec(Codec<T> primaryCodec, Codec<T> secondaryCodec, int prefixLength,
043          Predicate<byte[]> discriminator) {
044        Preconditions.checkNotNull(primaryCodec);
045        Preconditions.checkNotNull(secondaryCodec);
046        this.primaryCodec = primaryCodec;
047        this.secondaryCodec = secondaryCodec;
048        this.prefixLength = prefixLength;
049        this.discriminator = discriminator;
050      }
051    
052      /**
053       * Creates a new compatibility codec instance.
054       *
055       * @param primaryCodec the codec used to serialize objects, as well as deserialize them when the
056       *          first byte of the serialized format matches the discriminator.
057       * @param secondaryCodec the codec used to deserialize objects when the first byte of the
058       *          serialized format does not match the discriminator.
059       * @param prefixLength the length, in bytes, of the prefix of the message that is inspected for
060       *          determining the format.
061       * @param discriminator a predicate that will receive an array of at most prefixLength bytes
062       *          (it can receive less if the serialized format is shorter) and has to return true
063       *          if the primary codec should be used for deserialization, otherwise false.
064       */
065      public static <T> CompatibilityCodec<T> create(Codec<T> primaryCodec, Codec<T> secondaryCodec,
066          int prefixLength, Predicate<byte[]> discriminator) {
067        return new CompatibilityCodec<T>(primaryCodec, secondaryCodec, prefixLength, discriminator);
068      }
069    
070      @Override
071      public T deserialize(InputStream source) throws IOException {
072        final PushbackInputStream in = new PushbackInputStream(source, prefixLength);
073        final byte[] prefix = readAtMostPrefix(in);
074        in.unread(prefix);
075        return (discriminator.apply(prefix) ? primaryCodec : secondaryCodec).deserialize(in);
076      }
077    
078      private byte[] readAtMostPrefix(InputStream in) throws IOException {
079        final byte[] prefix = new byte[prefixLength];
080        int read = 0;
081        do {
082          final int readNow = in.read(prefix, read, prefixLength - read);
083          if (readNow == -1) {
084            byte[] newprefix = new byte[read];
085            System.arraycopy(prefix, 0, newprefix, 0, read);
086            return newprefix;
087          }
088          read += readNow;
089        } while (read < prefixLength);
090        return prefix;
091      }
092    
093      @Override
094      public void serialize(T item, OutputStream sink) throws IOException {
095        primaryCodec.serialize(item, sink);
096      }
097    }
