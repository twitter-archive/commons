017    package com.twitter.common.io;
018    
019    import com.google.common.base.Predicate;
020    import com.twitter.common.base.Closure;
021    
022    /**
023     * Encapsulates iteration over a typed data stream that can be filtered.
024     *
025     * @author John Sirois
026     */
027    public interface Streamer<T> {
028    
029      /**
030       * Processes a stream fully.  This may cause a database query to be executed, a file to be read
031       * or even just call {@link Iterable#iterator()} depending on the implementation.  Implementations
032       * guaranty that any resources allocated opening the stream will be closed whether or not process
033       * completes normally.
034       *
035       * @param work a closure over the work to be done for each item in the stream.
036       */
037      void process(Closure<T> work);
038    
039      /**
040       * Returns a {@code Streamer} that will process the same stream as this streamer, but will stop
041       * processing when encountering the first item for which {@code cond} is true.
042       *
043       * @param cond a predicate that returns {@code false} as long as the stream should keep being
044       *     processed.
045       * @return a streamer that will process items until the condition triggers.
046       */
047      Streamer<T> endOn(Predicate<T> cond);
048    
049      /**
050       * Returns a {@code Streamer} that will process the same stream as this streamer, but with any
051       * items failing the filter to be omitted from processing.
052       * @param filter a predicate that returns {@code true} if an item in the stream should be
053       *     processed
054       * @return a filtered streamer
055       */
056      Streamer<T> filter(Predicate<T> filter);
057    }
