018    package com.twitter.common.net.http.handlers;
019    
020    import com.google.common.base.Preconditions;
021    import com.google.inject.Inject;
022    import com.google.inject.name.Named;
023    
024    import javax.servlet.ServletException;
025    import javax.servlet.http.HttpServlet;
026    import javax.servlet.http.HttpServletRequest;
027    import javax.servlet.http.HttpServletResponse;
028    import java.io.IOException;
029    import java.io.PrintWriter;
030    import java.util.logging.Level;
031    import java.util.logging.Logger;
032    
033    /**
034     * A servlet that provides a way to remotely terminate the running process immediately.
035     *
036     * @author William Farner
037     */
038    public class AbortHandler extends HttpServlet {
039    
040      /**
041       * A {@literal @Named} binding key for the QuitHandler listener.
042       */
043      public static final String ABORT_HANDLER_KEY =
044          "com.twitter.common.net.http.handlers.AbortHandler.listener";
045    
046      private static final Logger LOG = Logger.getLogger(AbortHandler.class.getName());
047    
048      private final Runnable abortListener;
049    
050      /**
051       * Constructs a new AbortHandler that will notify the given {@code abortListener} when the servlet
052       * is accessed.  It is the responsibility of the listener to initiate an immediate shutdown of
053       * the system.
054       *
055       * @param abortListener Runnable to notify when the servlet is accessed.
056       */
057      @Inject
058      public AbortHandler(@Named(ABORT_HANDLER_KEY) Runnable abortListener) {
059        this.abortListener = Preconditions.checkNotNull(abortListener);
060      }
061    
062      @Override
063      protected void doGet(HttpServletRequest req, HttpServletResponse resp)
064          throws ServletException, IOException {
065        LOG.info(String.format("Received abort HTTP signal from %s (%s)",
066            req.getRemoteAddr(), req.getRemoteHost()));
067    
068        resp.setContentType("text/plain");
069        PrintWriter writer = resp.getWriter();
070        try {
071          writer.println("Aborting process NOW!");
072          writer.close();
073          abortListener.run();
074        } catch (Exception e) {
075          LOG.log(Level.WARNING, "Abort failed.", e);
076        }
077      }
078    }
