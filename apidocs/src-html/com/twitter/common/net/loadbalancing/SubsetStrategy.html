017    package com.twitter.common.net.loadbalancing;
018    
019    import com.google.common.base.Preconditions;
020    import com.google.common.collect.ImmutableSet;
021    import com.google.common.collect.Lists;
022    import com.google.common.collect.Sets;
023    import com.twitter.common.base.Closure;
024    import com.twitter.common.net.pool.ResourceExhaustedException;
025    import com.twitter.common.net.loadbalancing.RequestTracker.RequestResult;
026    
027    import java.util.Collection;
028    import java.util.Collections;
029    import java.util.List;
030    import java.util.Set;
031    
032    /**
033     * A load balancer that maintains a fixed upper bound on the number of backends that will be made
034     * available for a wrapped load balancer.
035     *
036     * TODO(William Farner): May want to consider periodically swapping subsets.
037     *
038     * TODO(William Farner): May want to catch ResourceExhaustedExceptions from wrapped strategy and adjust
039     *    subset if possible.
040     *
041     * @author William Farner
042     */
043    public class SubsetStrategy<S> implements LoadBalancingStrategy<S> {
044      private final LoadBalancingStrategy<S> wrapped;
045      private final int maxBackends;
046    
047      private Set<S> backendSubset = Sets.newHashSet();
048    
049      public SubsetStrategy(int maxBackends, LoadBalancingStrategy<S> wrapped) {
050        Preconditions.checkArgument(maxBackends > 0);
051        this.maxBackends = maxBackends;
052        this.wrapped = Preconditions.checkNotNull(wrapped);
053      }
054    
055      @Override
056      public void offerBackends(Set<S> offeredBackends, Closure<Collection<S>> onBackendsChosen) {
057        List<S> allTargets = Lists.newArrayList(offeredBackends);
058        Collections.shuffle(allTargets);
059        backendSubset = ImmutableSet.copyOf(
060            allTargets.subList(0, Math.min(maxBackends, allTargets.size())));
061        wrapped.offerBackends(backendSubset, onBackendsChosen);
062      }
063    
064      @Override
065      public void addConnectResult(S backendKey, ConnectionResult result,
066          long connectTimeNanos) {
067        if (backendSubset.contains(backendKey)) {
068          wrapped.addConnectResult(backendKey, result, connectTimeNanos);
069        }
070      }
071    
072      @Override
073      public void connectionReturned(S backendKey) {
074        if (backendSubset.contains(backendKey)) {
075          wrapped.connectionReturned(backendKey);
076        }
077      }
078    
079      @Override
080      public void addRequestResult(S requestKey, RequestResult result, long requestTimeNanos) {
081        Preconditions.checkNotNull(requestKey);
082    
083        if (backendSubset.contains(requestKey)) {
084          wrapped.addRequestResult(requestKey, result, requestTimeNanos);
085        }
086      }
087    
088      @Override
089      public S nextBackend() throws ResourceExhaustedException {
090        return wrapped.nextBackend();
091      }
092    }
