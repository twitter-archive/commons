017    package com.twitter.common.net.pool;
018    
019    import com.google.common.collect.ImmutableSet;
020    import com.google.common.collect.Sets;
021    
022    import java.util.Set;
023    
024    /**
025     * Util Class for dealing with dynamic sets of hosts.
026     *
027     * @author Florian Leibert
028     * @author Jake Mannix
029     */
030    public class DynamicHostSetUtil {
031    
032      /**
033       * Gets a snapshot of a set of dynamic hosts (e.g. a ServerSet) and returns a readable copy of
034       * the underlying actual endpoints.
035       * @param hostSet
036       * @param <T>
037       * @return
038       * @throws DynamicHostSet.MonitorException
039       */
040      public static <T> ImmutableSet<T> getSnapshot(DynamicHostSet<T> hostSet)
041          throws DynamicHostSet.MonitorException {
042        final Set<T> set = Sets.newHashSet();
043        hostSet.monitor(new DynamicHostSet.HostChangeMonitor<T>() {
044          @Override
045          public void onChange(ImmutableSet<T> hostSet) {
046            set.addAll(hostSet);
047          }
048        });
049        return ImmutableSet.copyOf(set);
050      }
051    }
