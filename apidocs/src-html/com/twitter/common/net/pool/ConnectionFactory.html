017    package com.twitter.common.net.pool;
018    
019    import com.twitter.common.quantity.Amount;
020    import com.twitter.common.quantity.Time;
021    
022    /**
023     * A factory for connections that also dictates policy for the size of the connection population.
024     *
025     * <p>TODO(John Sirois): separate concerns - mixing in willCreate/null protocol is already tangling
026     * implementation code
027     *
028     * @author John Sirois
029     */
030    public interface ConnectionFactory<S extends Connection<?, ?>> {
031    
032      /**
033       * Checks whether this factory might create a connection if requested.
034       *
035       * @return {@code} true if this factory might create a connection at this point in time; ie
036       * a call to {@link #create} might not have returned {@code null}.  May return true to multiple
037       * threads if concurrently creating connections.
038       */
039      boolean mightCreate();
040    
041      /**
042       * Attempts to create a new connection within the given timeout and subject to this factory's
043       * connection population size policy.
044       *
045       * @param timeout the maximum amount of time to wait
046       * @return a new connection or null if there are too many connections already
047       * @throws Exception if there was a problem creating the connection or establishing the connection
048       *     takes too long
049       */
050      S create(Amount<Long, Time> timeout) throws Exception;
051    
052      /**
053       * Destroys a connection.  It is an error to attempt to destroy a connection this factory did
054       * not {@link #create}
055       *
056       * @param connection The connection to destroy.
057       */
058      void destroy(S connection);
059    }
