017    package com.twitter.common.base;
018    
019    import com.google.common.collect.ImmutableList;
020    
021    /**
022     * Utility functions for working with exceptional functions.
023     *
024     * @author John Sirois
025     */
026    public final class ExceptionalFunctions {
027    
028      private ExceptionalFunctions() {
029        // utility
030      }
031    
032      /**
033       * Returns an {@link ExceptionalSupplier}/{@link java.util.concurrent.Callable} object that
034       * will return the result of {@code function} applied to {@code argument}.  Evaluation is lazy
035       * and un-memoized.
036       */
037      public static <S, T, E extends Exception> CallableExceptionalSupplier<T, E> curry(
038          final ExceptionalFunction<S, T, E> function, final S argument) {
039    
040        return new CallableExceptionalSupplier<T, E>() {
041          @Override public T get() throws E {
042            return function.apply(argument);
043          }
044        };
045      }
046    
047      /**
048       * Returns an ExceptionalFunction that is a composition of multiple ExceptionalFunctions.
049       */
050      public static <T, E extends Exception> ExceptionalFunction<T, T, E> compose(
051          final Iterable<ExceptionalFunction<T, T, E>> functions) {
052        return new ExceptionalFunction<T, T, E>() {
053          @Override public T apply(T input) throws E {
054            T result = input;
055            for (ExceptionalFunction<T, T, E> f : functions) {
056              result = f.apply(result);
057            }
058            return result;
059          }
060        };
061      }
062    
063      /**
064       * Returns a List of ExceptionalFunctions from variable number of ExceptionalFunctions.
065       */
066      public static <T, E extends Exception> ExceptionalFunction<T, T, E> compose(
067          ExceptionalFunction<T, T, E> function, ExceptionalFunction<T, T, E>... functions) {
068        return compose(ImmutableList.<ExceptionalFunction<T, T, E>>builder()
069            .add(function)
070            .add(functions)
071            .build());
072      }
073    }
