018    package com.twitter.common.collections;

020    import java.util.ArrayList;
021    import java.util.Arrays;
022    import java.util.Iterator;
023    import java.util.List;
024    import java.util.Map;
025    
026    import com.google.common.base.Function;
027    import com.google.common.collect.ImmutableList;
028    import com.google.common.collect.Iterables;
029    import com.google.common.collect.Lists;
030    import com.google.common.collect.MapMaker;

032    /**
033     * Utility functions for dealing with iterables.
034     *
035     * @author William Farner
036     */
037    public final class Iterables2 {
038    
039      private Iterables2() {
040        // Utility class.
041      }
042    
043      /**
044       * An iterator that zips multiple iterables into a single list iterator, filling missing values
045       * with a provided default.
046       *
047       * @param <T> The value type for the iterator.
048       */
049      private static class ZippingIterator<T> implements Iterator<List<T>> {
050    
051        private final Iterable<Iterable<T>> iterables;
052        private final T defaultValue;
053    
054        private List<Iterator<T>> iterators = null;
055        private final Map<Iterator<T>, Boolean> overflowing = new MapMaker().makeComputingMap(
056            new Function<Iterator<T>, Boolean>() {
057              @Override public Boolean apply(Iterator<T> iterator) { return false; }
058            });
059    
060        ZippingIterator(Iterable<Iterable<T>> iterables, T defaultValue) {
061          this.iterables = iterables;
062          this.defaultValue = defaultValue;
063        }
064    
065        private void init() {
066          if (iterators == null) {
067            // Iterables -> Iterators.
068            iterators = ImmutableList.copyOf(Iterables.transform(iterables,
069                new Function<Iterable<T>, Iterator<T>>() {
070                  @Override public Iterator<T> apply(Iterable<T> it) { return it.iterator(); }
071                }));
072          }
073        }
074    
075        @Override public boolean hasNext() {
076          init();
077          for (Iterator<T> it : iterators) {
078            if (it.hasNext()) {
079              return true;
080            }
081          }
082    
083          return false;
084        }
085    
086        @Override public List<T> next() {
087          init();
088          List<T> data = new ArrayList<T>(iterators.size());
089    
090          for (Iterator<T> it : iterators) {
091            if (it.hasNext()) {
092              data.add(it.next());
093            } else {
094              overflowing.put(it, true);
095              data.add(defaultValue);
096            }
097          }
098    
099          return data;
100        }
101    
102        @Override public void remove() {
103          init();
104          for (Iterator<T> it : iterators) {
105            if (!overflowing.get(it)) {
106              it.remove();
107            }
108          }
109        }
110    
111        @Override public String toString() {
112          return Lists.newArrayList(iterables).toString();
113        }
114      }
115    
116      /**
117       * Zips multiple iterables into one iterable that will return iterators to step over
118       * rows of the input iterators (columns).  The order of the returned values within each row will
119       * match the ordering of the input iterables. The iterators will iterate the length of the longest
120       * input iterable, filling other columns with {@code defaultValue}.
121       * The returned iterator is lazy, in that 'rows' are constructed as they are requested.
122       *
123       * @param iterables Columns to iterate over.
124       * @param defaultValue Default fill value when an input iterable is exhausted.
125       * @param <T> Type of value being iterated over.
126       * @return An iterator that iterates over rows of the input iterables.
127       */
128      public static <T> Iterable<List<T>> zip(final Iterable<Iterable<T>> iterables,
129          final T defaultValue) {
130    
131        return new Iterable<List<T>>() {
132          @Override public Iterator<List<T>> iterator() {
133            return new ZippingIterator<T>(iterables, defaultValue);
134          }
135        };
136      }
137    
138      /**
139       * Varargs convenience function to call {@link #zip(Iterable, Object)}.
140       *
141       * @param defaultValue Default fill value when an input iterable is exhausted.
142       * @param iterables Columns to iterate over.
143       * @param <T> Type of value being iterated over.
144       * @return An iterator that iterates over rows of the input iterables.
145       */
146      public static <T> Iterable<List<T>> zip(T defaultValue, Iterable<T>... iterables) {
147        return zip(Arrays.asList(iterables), defaultValue);
148      }
149    }
