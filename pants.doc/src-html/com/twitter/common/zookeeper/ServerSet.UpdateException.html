018    package com.twitter.common.zookeeper;
019    
020    import com.twitter.common.net.pool.DynamicHostSet;
021    import com.twitter.common.zookeeper.Group.JoinException;
022    import com.twitter.thrift.ServiceInstance;
023    import com.twitter.thrift.Status;
024    
025    import java.net.InetSocketAddress;
026    import java.util.Map;
027    
028    /**
029     * A logical set of servers registered in ZooKeeper.  Intended to be used by both servers in a
030     * common service and their clients.
031     *
032     * TODO(William Farner): Explore decoupling this from thrift.
033     *
034     * @author William Farner
035     */
036    public interface ServerSet extends DynamicHostSet<ServiceInstance> {
037    
038      /**
039       * Attempts to join a server set for this logical service group.
040       *
041       * @param endpoint the primary service endpoint
042       * @param additionalEndpoints and additional endpoints keyed by their logical name
043       * @param status the current service status
044       * @return an EndpointStatus object that allows the endpoint to adjust its status
045       * @throws JoinException if there was a problem joining the server set
046       * @throws InterruptedException if interrupted while waiting to join the server set
047       */
048      public EndpointStatus join(InetSocketAddress endpoint,
049          Map<String, InetSocketAddress> additionalEndpoints, Status status)
050          throws JoinException, InterruptedException;
051    
052      /**
053       * A handle to a service endpoint's status data that allows updating it to track current events.
054       */
055      public interface EndpointStatus {
056    
057        /**
058         * Attempts to update the status of the service endpoint associated with this endpoint.  If the
059         * {@code status} is {@link Status#DEAD} then the endpoint will be removed from the server set.
060         *
061         * @param status the current status of the endpoint
062         * @throws UpdateException if there was a problem writing the update
063         */
064        void update(Status status) throws UpdateException;
065      }
066    
067      /**
068       * Indicates an error updating a service's status information.
069       */
070      public static class UpdateException extends Exception {
071        public UpdateException(String message, Throwable cause) {
072          super(message, cause);
073        }
074      }
075    }
