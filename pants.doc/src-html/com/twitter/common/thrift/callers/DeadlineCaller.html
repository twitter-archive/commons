018    package com.twitter.common.thrift.callers;
019    
020    import com.google.common.base.Throwables;
021    import com.twitter.common.quantity.Amount;
022    import com.twitter.common.quantity.Time;
023    import com.twitter.common.thrift.TResourceExhaustedException;
024    import com.twitter.common.thrift.TTimeoutException;
025    import org.apache.thrift.async.AsyncMethodCallback;
026    
027    import javax.annotation.Nullable;
028    import java.lang.reflect.InvocationTargetException;
029    import java.lang.reflect.Method;
030    import java.util.concurrent.Callable;
031    import java.util.concurrent.ExecutionException;
032    import java.util.concurrent.ExecutorService;
033    import java.util.concurrent.Future;
034    import java.util.concurrent.RejectedExecutionException;
035    import java.util.concurrent.TimeoutException;
036    
037    /**
038     * A caller that imposes a time deadline on the underlying caller.  If the underlying calls fail
039     * to meet the deadline {@link TTimeoutException} is thrown.  If the executor service rejects
040     * execution of a task, {@link TResourceExhaustedException} is thrown.
041     *
042     * @author William Farner
043     */
044    public class DeadlineCaller extends CallerDecorator {
045      private final ExecutorService executorService;
046      private final Amount<Long, Time> timeout;
047    
048      /**
049       * Creates a new deadline caller.
050       *
051       * @param decoratedCaller The caller to decorate with a deadline.
052       * @param async Whether the caller is asynchronous.
053       * @param executorService The executor service to use for performing calls.
054       * @param timeout The timeout by which the underlying call should complete in.
055       */
056      public DeadlineCaller(Caller decoratedCaller, boolean async, ExecutorService executorService,
057          Amount<Long, Time> timeout) {
058        super(decoratedCaller, async);
059    
060        this.executorService = executorService;
061        this.timeout = timeout;
062      }
063    
064      @Override
065      public Object call(final Method method, final Object[] args,
066          @Nullable final AsyncMethodCallback callback,
067          @Nullable final Amount<Long, Time> connectTimeoutOverride) throws Throwable {
068        try {
069          Future<Object> result = executorService.submit(new Callable<Object>() {
070            @Override public Object call() throws Exception {
071              try {
072                return invoke(method, args, callback, null, connectTimeoutOverride);
073              } catch (Throwable t) {
074                Throwables.propagateIfInstanceOf(t, Exception.class);
075                throw new RuntimeException(t);
076              }
077            }
078          });
079    
080          try {
081            return result.get(timeout.getValue(), timeout.getUnit().getTimeUnit());
082          } catch (TimeoutException e) {
083            result.cancel(true);
084            throw new TTimeoutException(e);
085          } catch (ExecutionException e) {
086            throw Throwables.throwCause(e, false);
087          }
088        } catch (RejectedExecutionException e) {
089          throw new TResourceExhaustedException(e);
090        } catch (InvocationTargetException e) {
091          throw e.getCause();
092        }
093      }
094    }
