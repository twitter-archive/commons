018    package com.twitter.common.args.parsers;
019    
020    import java.util.List;
021    
022    import com.google.common.collect.ImmutableList;
023    
024    import com.twitter.common.args.Parsers;
025    import com.twitter.common.args.Parsers.Parser;
026    import com.twitter.common.collections.Pair;
027    
028    import static com.google.common.base.Preconditions.checkArgument;
029    import static com.twitter.common.args.Parsers.checkedGet;
030    
031    /**
032     * Pair parser.
033     *
034     * @author William Farner
035     */
036    public class PairParser extends TypeParameterizedParser<Pair> {
037    
038      public PairParser() {
039        super(Pair.class, 2);
040      }
041    
042      @Override
043      Pair doParse(String raw, List<Class<?>> paramParsers) {
044        final Parser leftParser = checkedGet(paramParsers.get(0));
045        final Parser rightParser = checkedGet(paramParsers.get(1));
046    
047        List<String> parts = ImmutableList.copyOf(Parsers.MULTI_VALUE_SPLITTER.split(raw));
048        checkArgument(parts.size() == 2,
049            "Only two values may be specified for a pair, you gave " + parts.size());
050    
051        return Pair.of(leftParser.parse(null, parts.get(0)),
052            rightParser.parse(null, parts.get(1)));
053      }
054    }
