018    package com.twitter.common.util;
019    
020    import com.google.common.annotations.VisibleForTesting;
021    import com.twitter.common.base.MorePreconditions;
022    
023    import java.io.InputStream;
024    import java.util.Properties;
025    import java.util.logging.Level;
026    import java.util.logging.Logger;
027    
028    /**
029     * Handles loading of a build properties file, and provides keys to look up known values in the
030     * properties.
031     *
032     * @author William Farner
033     */
034    public class BuildInfo {
035    
036      private static final Logger LOG = Logger.getLogger(BuildInfo.class.getName());
037    
038      private static final String DEFAULT_BUILD_PROPERTIES_PATH = "build.properties";
039    
040      private final String resourcePath;
041    
042      private Properties properties = null;
043    
044      /**
045       * Creates a build info container that will use the default properties file path.
046       */
047      public BuildInfo() {
048        this(DEFAULT_BUILD_PROPERTIES_PATH);
049      }
050    
051      /**
052       * Creates a build info container, reading from the given path.
053       *
054       * @param resourcePath The resource path to read build properties from.
055       */
056      public BuildInfo(String resourcePath) {
057        this.resourcePath = MorePreconditions.checkNotBlank(resourcePath);
058      }
059    
060      @VisibleForTesting
061      public BuildInfo(Properties properties) {
062        this.resourcePath = null;
063        this.properties = properties;
064      }
065    
066      private void fetchProperties() {
067        properties = new Properties();
068        LOG.info("Fetching build properties from " + resourcePath);
069        InputStream in = ClassLoader.getSystemResourceAsStream(resourcePath);
070        if (in == null) {
071          LOG.warning("Failed to fetch build properties from " + resourcePath);
072          return;
073        }
074    
075        try {
076          properties.load(in);
077        } catch (Exception e) {
078          LOG.log(Level.WARNING, "Failed to load properties file " + resourcePath, e);
079        }
080      }
081    
082      /**
083       * Fetches the properties stored in the resource location.
084       *
085       * @return The loaded properties, or a default properties object if there was a problem loading
086       *    the specified proeprites resource.
087       */
088      public Properties getProperties() {
089        if (properties == null) fetchProperties();
090        return properties;
091      }
092    
093      /**
094       * Values of keys that are expected to exist in the loaded properties file.
095       */
096      public enum Key {
097        PATH("build.path"),
098        USER("build.user.name"),
099        MACHINE("build.machine"),
100        DATE("build.date"),
101        TIME("build.time"),
102        TIMESTAMP("build.timestamp"),
103        GIT_TAG("build.git.tag"),
104        GIT_REVISION("build.git.revision"),
105        GIT_BRANCHNAME("build.git.branchname");
106    
107        public final String value;
108        private Key(String value) {
109          this.value = value;
110        }
111      }
112    }
