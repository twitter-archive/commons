018    package com.twitter.common.util;
019    
020    import com.google.common.base.Function;
021    import com.google.common.base.Preconditions;
022    import com.twitter.common.stats.Stats;
023    import com.twitter.common.memcached.Memcached;
024    import net.spy.memcached.MemcachedClient;
025    
026    import java.util.List;
027    import java.util.concurrent.atomic.AtomicLong;
028    import java.util.logging.Level;
029    import java.util.logging.Logger;
030    
031    /**
032     * Class responsible for pulling work items off of a kestrel queue.
033     *
034     * @author William Farner
035     */
036    public class KestrelConsumer {
037      private static Logger log = Logger.getLogger(KestrelConsumer.class.getName());
038    
039      private MemcachedClient queue;
040      private final List<String> kestrelServers;
041      private final String queueName;
042      private final Function<String, Boolean> taskHandler;
043    
044      // Controls backoffs when there is no work available.
045      private static final int MIN_DELAY_MS = 1;
046      private static final int MAX_BACKOFF_DELAY_MS = 8192;
047      private int backoffDelayMs = MIN_DELAY_MS;
048    
049      // Tracks rate at which items are being removed from the queue.
050      private final AtomicLong stats = Stats.exportLong("kestrel_consumption");
051    
052      /**
053       * Creates a new kestrel consumer that will communicate with the given kestrel servers (where
054       * a server string is formatted as host:port).
055       *
056       * @param kestrelServers The kestrel servers to pull work from.
057       * @param queueName The name of the kestrel queue to pull work from.
058       * @param taskHandler The handler for new work retrieved from the kestrel queue. The handler
059       *    should return {@code false} if the work item was not successfully handled.
060       */
061      public KestrelConsumer(List<String> kestrelServers, String queueName,
062          Function<String, Boolean> taskHandler) {
063        Preconditions.checkNotNull(kestrelServers);
064        Preconditions.checkNotNull(queueName);
065        Preconditions.checkNotNull(taskHandler);
066        this.kestrelServers = kestrelServers;
067        this.queueName = queueName;
068        this.taskHandler = taskHandler;
069      }
070    
071      public void initialize() {
072        queue = Memcached.newKestrelClient(kestrelServers);
073      }
074    
075      public void consumeForever() {
076        Preconditions.checkNotNull(queue);
077    
078        log.info("Consuming items from the kestrel queue forever.");
079        while(true) {
080          Object task = null;
081          while (task == null) {
082            try {
083              Thread.sleep(backoffDelayMs);
084            } catch (InterruptedException e) {
085              log.log(Level.INFO, "Interrupted while sleeping.", e);
086            }
087    
088            task = queue.get(queueName);
089    
090            if (task == null) {
091              // Back off exponentially (capped).
092              backoffDelayMs = Math.min(backoffDelayMs * 2, MAX_BACKOFF_DELAY_MS);
093              log.info("No work, backing off for " + backoffDelayMs + " ms.");
094            } else {
095              stats.incrementAndGet();
096              backoffDelayMs = MIN_DELAY_MS;
097            }
098          }
099    
100          if (task instanceof String) {
101            taskHandler.apply((String) task);
102          } else {
103            log.severe("Unexpected task " + task + " of type " + task.getClass().getName());
104          }
105        }
106      }
107    }
