035      /**
036       * Given an url host, tokenizes the host parts by dot, reverses them, and pads (with "") or
037       * truncates up to numParts pieces.  Truncation happens post token reversal.  This method properly
038       * accounts for two-part top-level domains as listed at
039       * http://www.surbl.org/two-level-tlds as of 05/12/2010.  For example, www.twitter.co.uk will
040       * be returned as  ["co.uk", "twitter", "www"].
041       *
042       * @param host A url host
043       * @param numParts The number of host parts to return.
044       * @return The reversed and tokenized list of host parts from host.
045       */
046      public static List<String> getReversedDomainParts(String host, int numParts) {
047        List<String> domainParts = Lists.newArrayList();
048        for (String item : host.split("\\.")) {
049          domainParts.add(item);
050        }
051        Collections.reverse(domainParts);
052    
053        /*
054         * If the result is at least length two, see if we can squash together the last two
055         * parts as a top-level domain.
056         */
057        if (domainParts.size() >= 2 && TWO_PART_TLDS.contains(
058            Pair.of(domainParts.get(1), domainParts.get(0)))) {
059          domainParts.set(0, domainParts.get(1) + "." + domainParts.get(0));
060          domainParts.remove(1);
061        }
062    
063        if (domainParts.size() > numParts) {
064          return domainParts.subList(0, numParts);
065        } else if (domainParts.size() < numParts) {
066          int diff = numParts - domainParts.size();
067          for (int i = 0; i < diff; ++i) {
068            domainParts.add("");
069          }
070        }
071    
072        return domainParts;
073      }
