018    package com.twitter.common.net.pool;
019    
020    import com.twitter.common.quantity.Amount;
021    import com.twitter.common.quantity.Time;
022    
023    /**
024     * A factory for connections that also dictates policy for the size of the connection population.
025     *
026     * <p>TODO(John Sirois): separate concerns - mixing in willCreate/null protocol is already tangling
027     * implementation code
028     *
029     * @author John Sirois
030     */
031    public interface ConnectionFactory<S extends Connection<?, ?>> {
032    
033      /**
034       * Checks whether this factory might create a connection if requested.
035       *
036       * @return {@code} true if this factory might create a connection at this point in time; ie
037       * a call to {@link #create} might not have returned {@code null}.  May return true to multiple
038       * threads if concurrently creating connections.
039       */
040      boolean mightCreate();
041    
042      /**
043       * Attempts to create a new connection within the given timeout and subject to this factory's
044       * connection population size policy.
045       *
046       * @param timeout the maximum amount of time to wait
047       * @return a new connection or null if there are too many connections already
048       * @throws Exception if there was a problem creating the connection or establishing the connection
049       *     takes too long
050       */
051      S create(Amount<Long, Time> timeout) throws Exception;
052    
053      /**
054       * Destroys a connection.  It is an error to attempt to destroy a connection this factory did
055       * not {@link #create}
056       *
057       * @param connection The connection to destroy.
058       */
059      void destroy(S connection);
060    }
