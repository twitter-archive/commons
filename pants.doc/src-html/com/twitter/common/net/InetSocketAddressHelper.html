018    package com.twitter.common.net;
019    
020    import com.google.common.base.Function;
021    import com.google.common.base.Preconditions;
022    import com.google.common.collect.Iterables;
023    import com.google.common.collect.Sets;
024    import org.apache.commons.lang.StringUtils;
025    
026    import java.net.InetAddress;
027    import java.net.InetSocketAddress;
028    import java.net.UnknownHostException;
029    import java.util.Set;
030    
031    /**
032     * A utility that can parse [host]:[port] pairs or :[port] designators into instances of
033     * {@link java.net.InetSocketAddress}. The literal '*' can be specified for port as an alternative
034     * to '0' to indicate any local port.
035     *
036     * @author John Sirois
037     */
038    public final class InetSocketAddressHelper {
039    
040      /**
041       * A function that uses {@link #parse(String)} to map an endpoint spec to an
042       * {@link InetSocketAddress}.
043       */
044      public static final Function<String, InetSocketAddress> STR_TO_INET =
045          new Function<String, InetSocketAddress>() {
046            @Override public InetSocketAddress apply(String value) {
047              return parse(value);
048            }
049          };
050    
051      public static final Function<InetSocketAddress, String> INET_TO_STR =
052          new Function<InetSocketAddress, String>() {
053            @Override public String apply(InetSocketAddress addr) {
054              return InetSocketAddressHelper.toString(addr);
055            }
056          };
057    
058      /**
059       * Attempts to parse an endpoint spec into an InetSocketAddress.
060       *
061       * @param value the endpoint spec
062       * @return a parsed InetSocketAddress
063       * @throws NullPointerException     if {@code value} is {@code null}
064       * @throws IllegalArgumentException if {@value} cannot be parsed
065       */
066      public static InetSocketAddress parse(String value) {
067        Preconditions.checkNotNull(value);
068    
069        String[] spec = value.split(":", 2);
070        if (spec.length != 2) {
071          throw new IllegalArgumentException("Invalid socket address spec: " + value);
072        }
073    
074        String host = spec[0];
075        int port = asPort(spec[1]);
076    
077        return StringUtils.isEmpty(host)
078            ? new InetSocketAddress(port)
079            : InetSocketAddress.createUnresolved(host, port);
080      }
081    
082      /**
083       * Attempts to return a usable String given an InetSocketAddress.
084       *
085       * @param value the InetSocketAddress.
086       * @return the String representation of the InetSocketAddress.
087       */
088      public static String toString(InetSocketAddress value) {
089        Preconditions.checkNotNull(value);
090        return value.getHostName() + ":" + value.getPort();
091      }
092    
093      private static int asPort(String port) {
094        if ("*".equals(port)) {
095          return 0;
096        }
097        try {
098          return Integer.parseInt(port);
099        } catch (NumberFormatException e) {
100          throw new IllegalArgumentException("Invalid port: " + port, e);
101        }
102      }
103    
104      public static InetSocketAddress getLocalAddress(int port) throws UnknownHostException {
105        String ipAddress = InetAddress.getLocalHost().getHostAddress();
106        return new InetSocketAddress(ipAddress, port);
107      }
108    
109      private InetSocketAddressHelper() {
110        // utility
111      }
112    
113      /**
114       * Converts backend definitions (in host:port form) a set of socket addresses.
115       *
116       * @param backends Backends to convert.
117       * @return Sockets representing the provided backends.
118       */
119      public static Set<InetSocketAddress> convertToSockets(Iterable<String> backends) {
120        return Sets.newHashSet(Iterables.transform(backends, STR_TO_INET));
121      }
122    }
